class Solution {
public:
    int maximumSum(vector<int>& nums) {
        
        map<int, int> mp;
        
        int ans = -1;
        
        for(int x: nums) {
            int sum = 0;
            int tmp = x;
            while(tmp > 0) {
                sum += (tmp % 10);
                tmp /= 10;
            }
            if(mp.count(sum)) {
                ans = max(ans, x + mp[sum]);
                mp[sum] = max(x, mp[sum]);
            } else {
                mp[sum] = x;
            }
        }
        
        return ans;
    }
};
