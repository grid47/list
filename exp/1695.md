class Solution {
public:
    int maximumUniqueSubarray(vector<int>& nums) {
        
        int sum = 0, n = nums.size();
        map<int, int> mp;
        
        int mx = 0;
        int j = 0;
        for(int i = 0; i < n; i++) {
            mp[nums[i]]++;
            sum += nums[i];
            
            while(mp[nums[i]] > 1) {
                mp[nums[j]]--;
                sum -= nums[j];
                j++;
            }
            
            mx = max(mx, sum);
        }
        return mx;
    }
};
