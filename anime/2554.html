class Solution { public: int n; unordered_map mp; int maxCount(vector& ban, int n, int mxS) { // this->n = n; for(int x: ban) mp[x] = true; // memset(memo, -1, sizeof(memo)); // memo.resize(n + 1); // return dp(1, mxS); int cnt = 0; for(int i = 1; i < n + 1; i++) { if(!mp.count(i) && mxS >= i) { cnt++; mxS -= i; } if(mxS < 0) break; } return cnt; } };
