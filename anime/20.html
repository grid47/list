class Solution { public: bool isValid(string s) { stack stk; for(char x: s) { if(x == '(' || x == '{' || x == '[') { stk.push(x); } else { if(x == ')' && !stk.empty() && stk.top() == '(') stk.pop(); else if(x == '}' && !stk.empty() && stk.top() == '{') stk.pop(); else if(x == ']' && !stk.empty() && stk.top() == '[') stk.pop(); else return false; } } return stk.empty(); } };
