class Solution { public: long long matrixSumQueries(int n, vector>& q) { long long res = 0; int seen[2][10001] = {}; int cnt[2] = {n, n}; for(int i = q.size() - 1; i >= 0; i--) { int type = q[i][0], id = q[i][1], val = q[i][2]; if(!seen[type][id]) { seen[type][id] = 1; res += val * cnt[!type]; --cnt[type]; } } return res; } };
