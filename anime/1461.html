class Solution { public: bool hasAllCodes(string s, int k) { int n = s.size(); set cnt; int tmp = 0; for(int i = 0; i < n; i++) { if(i < k) { tmp = tmp * 2 + (s[i] == '0'? 0: 1); continue; } cnt.insert(tmp); tmp *= 2; tmp &= ((1 << k) - 1); tmp += (s[i] == '0'? 0: 1); } cnt.insert(tmp); // cout << cnt.size() << " " << (1 << k); return cnt.size() == (1 << k); } };
