class Solution { public: void gen(string op, vector &ans, int open, int closed, int n, int i) { if(i== 2*n) { ans.push_back(op); return; } if(open < n) { gen(op + "(", ans, open + 1, closed,n, i + 1); } if(open > closed) { gen(op + ")", ans, open, closed + 1,n, i + 1); } } vector generateParenthesis(int n) { vector ans; gen("", ans, 0, 0, n, 0); return ans; } };
