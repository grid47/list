class Solution { public: vector row, col; vector> vis, vs; int m, n; int minCost(vector& start, vector& home, vector& row, vector& col) { m = row.size(); n = col.size(); int ans = 0; int i = start[0], x = home[0]; int j = start[1], y = home[1]; while(i != x) { i += (x - i) / abs(x - i); ans += row[i]; } while(j != y) { j += (y - j) / abs(y - j); ans += col[j]; } return ans; } };
