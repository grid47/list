class Solution { public: string intToRoman(int num) { vector nums = { 1000, 900, 500, 400, 100, 90, 50, 40, 10, 9, 5, 4, 1 }; vector chrs = { "M", "CM", "D", "CD", "C", "XC", "L", "XL", "X", "IX", "V", "IV", "I" }; int idx = 0; string res = ""; while(num > 0) { while(num >= nums[idx]) { num -= nums[idx]; res += chrs[idx]; } idx++; } return res; } };
