class Solution { public: int n; vector> combine(int n, int k) { this->n = n; vector> ans; vector tmp; bt(ans, 1, k, tmp); return ans; } void bt(vector> &ans, int i, int k, vector &tmp) { if(tmp.size() == k) { ans.push_back(tmp); return; } if(i == n + 1) return; for(int j = i; j <= n; j++) { tmp.push_back(j); bt(ans, j + 1, k, tmp); tmp.pop_back(); } } };
