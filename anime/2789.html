class Solution { public: long long maxArrayValue(vector& in) { int n = in.size(); vector nums(n, 0); for(int i = 0; i < n; i++) nums[i] = in[i]; long long res = nums[n - 1]; for(int i = n - 1; i >= 0; i--) { res = max(res, (long long)nums[i]); if(i > 0 && nums[i] >= nums[i - 1]) { nums[i - 1] += nums[i]; } } return res; } };
