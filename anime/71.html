class Solution { public: string simplifyPath(string path) { set dot = {"..", ".", ""}; string res = "", tmp; stringstream ss(path); stack stk; while(getline(ss, tmp, '/')) { if(tmp == ".." && !stk.empty()) stk.pop(); else if (!dot.count(tmp)) stk.push(tmp); } while(!stk.empty()) { res = "/" + stk.top() + res; stk.pop(); } return res == ""? "/": res; } };
