class Solution { public: vector> permute(vector& nums) { map mp; vector> ans; vector tmp; bt(ans, tmp, nums, 0, mp); return ans; } void bt(vector> &ans, vector &tmp, vector &nums, int idx, map &mp) { if(idx == nums.size()) { ans.push_back(tmp); return; } for(int i = 0; i < nums.size(); i++) { if(mp.count(nums[i])) continue; mp[nums[i]] = true; tmp.push_back(nums[i]); bt(ans, tmp, nums, idx + 1, mp); tmp.pop_back(); mp.erase(nums[i]); } } };
