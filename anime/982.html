class Solution { public: int countTriplets(vector& nums) { int n = nums.size(); int res = 0; vector mp(1<<16, 0); for(int i = 0; i < n; i++) for(int j = 0; j < n; j++) { mp[nums[i] & nums[j]]++; } for(int i = 0; i < n; i++) for(int j = 0; j < (1 << 16); j++) if((j & nums[i]) == 0) res += mp[j]; return res; } };
