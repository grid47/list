class Solution { public: int minimumDeletions(vector& nums) { int xi = 0, ni = 0; int n = nums.size(); for(int i = 0; i < n; i++) { if(nums[xi] < nums[i]) xi = i; if(nums[ni] > nums[i]) ni = i; } if(xi == ni) return 1; int mx = (xi < ni)? ni: xi; int mn = (xi < ni)? xi: ni; int ans = (mx - 0 + 1); ans = min(ans, (n - mn)); ans = min(ans, (n - mx) + (mn - 0 + 1)); return ans; } };
