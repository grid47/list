class Solution { public: int expressiveWords(string s, vector& words) { int res = 0; for(auto &w : words) if(check(s, w)) res++; return res; } bool check(string s, string w) { int n = s.size(), m = w.size(), j = 0; for(int i = 0; i < n; i++) if(j < m && s[i] == w[j]) j++; else if (i > 1 && s[i - 2] == s[i - 1] && s[i - 1] == s[i]); else if (i > 0 && i < n - 1 && s[i -1] == s[i] && s[i] == s[i +1]); else return false; return j == m; } };
