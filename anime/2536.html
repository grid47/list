class Solution { public: vector> rangeAddQueries(int n, vector>& q) { vector> ans(n, vector(n, 0)); for(int i = 0; i < q.size(); i++) { auto it = q[i]; for(int j = q[i][0]; j <= q[i][2]; j++) { ans[j][q[i][1]] += 1; if(q[i][3] < n - 1) ans[j][q[i][3] + 1] -=1; } } for(int i = 0; i < n; i++) for(int j = 1; j < n; j++) ans[i][j] += ans[i][j - 1]; return ans; } };
