class Solution { public: vector spiralOrder(vector>& mtx) { int cBgn = 0, cEnd = mtx[0].size() - 1; int rBgn = 0, rEnd = mtx.size() - 1; vector ans; while(cBgn <= cEnd && rBgn <= rEnd) { for(int i = cBgn; i <= cEnd; i++) ans.push_back(mtx[rBgn][i]); rBgn++; for(int i = rBgn; i <= rEnd; i++) ans.push_back(mtx[i][cEnd]); cEnd--; if(rBgn <= rEnd) for(int i = cEnd; i >= cBgn; i--) ans.push_back(mtx[rEnd][i]); rEnd--; if(cBgn <= cEnd) for(int i = rEnd; i >= rBgn; i--) ans.push_back(mtx[i][cBgn]); cBgn++; } return ans; } };
