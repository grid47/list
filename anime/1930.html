class Solution { public: int countPalindromicSubsequence(string num) { int n = num.size(), res = 0; vector fst(26,n), lst(26,0); for(int i = 0; i < n; i++) { fst[num[i]-'a']= min(fst[num[i]-'a'], i); lst[num[i]-'a'] = i; } for(int i = 0; i < 26; i++) { if(fst[i] < lst[i]) res += unordered_set(num.begin() + fst[i] + 1, num.begin() + lst[i]).size(); } return res; } };
