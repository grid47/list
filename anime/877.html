class Solution { public: vector p; int n; map> memo; bool dp(int i, int j) { if(i == j) return p[i]; if(memo.count(i) && memo[i].count(j)) return memo[i][j]; return memo[i][j] = max(p[i] - dp(i + 1, j), p[j] - dp(i, j - 1)) ; } bool stoneGame(vector& piles) { this->p = piles; n = piles.size(); return dp(0, n - 1) >= 0; } };
