class Solution { public: int minOperations(string s) { int zero = 0, one = 0, zf = false, of = false; for(int i = 0; i < s.size(); i++) { if(zf && s[i] == '0') zero++; if(!zf && s[i] == '1') zero++; if(!of && s[i] == '0') one++; if(of && s[i] == '1') one++; zf = !zf, of = !of; } return min(zero, one); } };
