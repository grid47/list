/** * Definition for singly-linked list. * struct ListNode { * int val; * ListNode *next; * ListNode() : val(0), next(nullptr) {} * ListNode(int x) : val(x), next(nullptr) {} * ListNode(int x, ListNode *next) : val(x), next(next) {} * }; */ class Solution { public: ListNode* deleteMiddle(ListNode* head) { if(!head->next) return nullptr; ListNode* slw = head, *fst = head->next->next; while(fst && fst->next) { fst = fst->next->next; slw = slw->next; } slw->next = slw->next->next; return head; } };
