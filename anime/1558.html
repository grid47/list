class Solution { public: int minOperations(vector& nums) { int cnt = 0; int n = nums.size(); int zeros = 0; while(zeros != n) { for(int i = 0; i < n; i++) { if(nums[i] & 1) cnt++, nums[i]--; } zeros = 0; int ch = true; for(int i = 0; i < n; i++) { if(nums[i] > 0 && ch) cnt++, ch = false; nums[i] /= 2; if(nums[i] == 0) zeros++; } } return cnt; } };
