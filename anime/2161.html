class Solution { public: vector pivotArray(vector& nums, int pivot) { int low = 0, same = 0, high; for(auto num: nums) { if(num < pivot) low++; else if(num == pivot) same++; } high = low + same; same = low; low = 0; vector res(nums.size(), 0); for(auto num: nums) { if(num < pivot) res[low++] = num; else if(num == pivot) res[same++] = num; else res[high++] = num; } return res; } };
