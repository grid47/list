class Solution { public: int xorAllNums(vector& nums1, vector& nums2) { int m = nums1.size(), n = nums2.size(); int xr1 = nums1[0], xr2 = nums2[0]; for(int i = 1; i < m; i++) xr1 ^= nums1[i]; for(int i = 1; i < n; i++) xr2 ^= nums2[i]; if((n % 2) == 0) { if((m % 2) == 0) { return 0; } else { return xr2; } } else { if((m % 2) == 0) { return xr1; } else { return xr1 ^ xr2; } } return -1; } };
