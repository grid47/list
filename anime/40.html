class Solution { public: map, bool> mp; vector> combinationSum2(vector& cand, int target) { vector> ans; vector tmp; mp.clear(); sort(cand.begin(), cand.end()); bt(ans, cand, 0, target, tmp); return ans; } void bt(vector> &ans, vector &nums, int idx, int sum, vector &tmp) { if(sum == 0) { ans.push_back(tmp); return; } if(sum < 0 || idx == nums.size()) return; for(int i = idx; i < nums.size(); i++) { if(i && i > idx && nums[i] == nums[i-1]) continue; tmp.push_back(nums[i]); bt(ans, nums, i + 1, sum - nums[i], tmp); tmp.pop_back(); } } };
