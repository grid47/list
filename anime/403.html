class Solution { public: bool canCross(vector& stones) { int n = stones.size(); map> mp; mp[0].insert(1); for(int i = 0; i < n; i++) { int st = stones[i]; for(int step : mp[st]) { int tgt = st + step; if(tgt == stones[stones.size() - 1]) return true; mp[tgt].insert(step + 1); mp[tgt].insert(step); if(step - 1 > 0) mp[tgt].insert(step - 1); } } return false; } };
