class Solution { public: int rob(vector& nums) { int n = nums.size(); if(n == 1) return nums[0]; vector dp(n, 0); dp[0] = 0; dp[1] = nums[0]; for(int i = 2; i < n; i++) dp[i] = max(dp[i - 1], dp[i - 2] + nums[i - 1]); int r1 = dp[n - 1]; dp.resize(n + 1, 0); dp[1] = 0; dp[2] = nums[1]; for(int i = 3; i < n + 1; i++) dp[i] = max(dp[i - 1], dp[i - 2] + nums[i - 1]); return max(dp[n], r1); } };
