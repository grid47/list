class Solution { public: vector circularGameLosers(int n, int k) { vector rcvd(n, 0); int i = 0; int cnt = 1; rcvd[0] = 1; while(rcvd[i] == 1) { i = (i + cnt * k) % n; if(rcvd[i] == 1) break; rcvd[i] = 1; cnt++; } vector ans; for(int j = 0; j < n; j++) { // cout << rcvd[j] << " "; if(rcvd[j] == 0) ans.push_back(j + 1); } return ans; } };
