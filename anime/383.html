class Solution { public: bool canConstruct(string ransomNote, string magazine) { map mp; for(int x: magazine) mp[x]++; for(int j = 0; j < ransomNote.size(); j++) { if(!mp.count(ransomNote[j])) return false; else { mp[ransomNote[j]]--; if(mp[ransomNote[j]] == 0) mp.erase(ransomNote[j]); } } return true; } };
