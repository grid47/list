class Solution { public: int maximumStrongPairXor(vector& nums) { sort(nums.begin(), nums.end()); int n = nums.size(), res = 0; for(int i = 0; i < n; i++) for(int j = i; j < n; j++) { if((nums[j] - nums[i] <= nums[i]) && (res <= (nums[i] ^ nums[j]))){ // cout << nums[i] << " " << nums[j] << "\n"; res = max(res, nums[i] ^ nums[j]); } } return res; } };
