class Solution { TreeNode *prv = NULL, * fst = NULL , *scd = NULL; void inorder(TreeNode* node) { if(!node) return; inorder(node->left); if (prv && node->val < prv->val) { if (!fst) fst = prv; scd = node; } prv = node; inorder(node->right); } public: void recoverTree(TreeNode* node) { inorder(node); swap(fst->val, scd->val); } };
