class Solution { public: int minOperations(vector& nums) { int cnt = 0; for(int x: nums) if (x == 1) cnt++; if(cnt > 0) return nums.size() - cnt; int t = INT_MAX; for(int i = 0; i < nums.size(); i++) { int last = nums[i]; for(int j = i + 1; j < nums.size(); j++) { last = __gcd(last, nums[j]); if(last == 1) { t = min(t, j - i); } } } return t == INT_MAX? -1: t + nums.size() - 1; } };
