int countPalindromicSubsequence(string num) {
    int n = num.size(), res = 0;        
    vector<int> fst(26,n), lst(26,0);
    for(int i = 0; i < n; i++) {
        fst[num[i]-'a']= min(fst[num[i]-'a'], i);
        lst[num[i]-'a'] = i;
    }
    for(int i = 0; i < 26; i++) {
if(fst[i] < lst[i]) res += unordered_set<char>(num.begin() + fst[i] + 1, num.begin() + lst[i]).size();
    }
    return res;        
}

bool areOccurrencesEqual(string s) {
    int cnt[26] = {}, m_cnt = 0;
    for (auto ch : s)
        m_cnt = max(m_cnt, ++cnt[ch - 'a']);
    return all_of(begin(cnt), end(cnt), [&m_cnt](int t){ return t == 0 || t == m_cnt; });
}

int maxCompatibilitySum(vector<vector<int>>& students, vector<vector<int>>& mentors) {
    vector<int> pos;
    int ans = 0;
    for(int i = 0; i < students.size(); i++)
        pos.push_back(i);
    do {

        int curr = 0;
        for(int i = 0; i < students.size(); i++)
        for(int j = 0; j < students[pos[i]].size(); j++)
            curr += (students[pos[i]][j] == mentors[i][j]);
        ans = max(ans, curr);

    } while(next_permutation(pos.begin(), pos.end()));
    return ans;
}

• A prefix sum approach is optimal for this problem since it allows us to efficiently check each index with a running sum calculation.
Steps:
• Calculate the total sum of the array.
• Iterate through each index and check if the sum of elements to the left equals the sum of elements to the right by comparing the prefix sum with the total sum.
Empty Inputs:
• The input will always contain at least one element, so no need to handle an empty array.
Large Inputs:
• The algorithm is efficient enough for arrays of length up to 100.
Special Values:
• The solution should handle arrays with negative numbers and arrays where all elements are the same.
Constraints:
• Ensure the solution runs efficiently for the maximum input size.
int findMiddleIndex(vector<int>& nums) {
    int n = nums.size();
    vector<int> sum(n + 1, 0);
    for(int i = 0; i < n; i++)
        sum[i + 1] = nums[i] + sum[i];
    
    int net = sum[n];
    
    for(int i = 0; i < n; i++) {
        if(net - sum[i + 1] == sum[i]) return i;
    }
    return -1;
}
