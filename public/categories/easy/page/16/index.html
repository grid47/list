Leetcode 1232: Check If It Is a Straight Line

long long findScore(vector<int>& nums) {
    long long score = 0;
    priority_queue<vector<int>, vector<vector<int>>, greater<vector<int>>> pq;
    int n = nums.size();
    unordered_map<int, int> mp;
    for(int i = 0; i < n; i++)
        pq.push({nums[i], i});
    
    while(!pq.empty()) {
        while(!pq.empty() && mp.count(pq.top()[1])) pq.pop();
        if(!pq.empty()) {
            score += pq.top()[0];
            mp[pq.top()[1]] = true;
            mp[pq.top()[1] + 1] = true;
            mp[pq.top()[1] - 1] = true;                
            pq.pop();
        }
    }
    return score;
}

vector<int> colorTheArray(int n, vector<vector<int>>& queries) {
int c[100003] = {}, adj = 0;
vector<int> res;
for (const auto &q : queries) {
    int i = q[0] + 1, color_i = q[1];
    if (c[i])
        adj -= (c[i - 1] == c[i]) + (c[i + 1] == c[i]);
    c[i] = color_i;
    adj += (c[i - 1] == c[i]) + (c[i + 1] == c[i]);
    res.push_back(adj);
}
return res;
}

Leetcode 1287: Element Appearing More Than 25% In Sorted Array

int longestSemiRepetitiveSubstring(string s) {
    
    vector<int> sem(10, 0);
    int cnt = 0;
    
    int j = 0, len = 1;        
    for(int i = 1; i < s.size(); i++) {
        if(s[i] == s[i - 1]) {
            cnt++;
            sem[s[i] - '0']++;
        }
        while(cnt > 1 && j < i) {
            if(s[j] == s[j + 1]) {
                sem[s[j] - '0']--;
                cnt--;
            }
            j++;
        }
        len = max(len, i - j + 1);
    }
    return len;
}

You can choose to move left or right when the character is ‘’, and the goal is to calculate the maximum distance from the origin you can reach after completing all the moves.

int furthestDistanceFromOrigin(string moves) {
    int n=moves.length();
    
    int l=0, r=0;
    for(int i=0;i<n;i++){
        if(moves[i]=='L' || moves[i]=='_'){
            l--;
        }else{
            l++;
        }
        
        if(moves[i]=='R' || moves[i]=='_'){
            r++;
        }else{
            r--;
        }
    }
    
    if(l<0) l *= -1;
    if(r<0) r *= -1;
		
    return max(l,r);
}

Approach: The solution involves iterating through the list of students, sorting the thresholds, and using a simple check to count the valid ways to form groups where all students are happy.

int countWays(vector<int>& nums) {
    sort(nums.begin(),nums.end());
    int ans = 0, n = nums.size();
    int selected = 0;
    if(nums[0]!=0) ans = 1; // Not Selecting AnyOne
    for(int i=0;i<n;i++) {
        selected++;
        if(selected>nums[i]) {  // No. of Selected Students is strictly greater than nums[i].
            if(i+1<n && selected<nums[i+1]) // Considering from (i+1) to n Students is not Selected.
                ans++; 
            else if(i+1==n) ans++;  // Last Student
        }
    }
    return ans;
