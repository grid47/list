Leetcode 2352: Equal Row and Column Pairs

Code:

class Solution {
public:
    int equalPairs(vector<vector<int>>& grid) {
        int n = grid.size();
        int ans = 0;
        for (int c = 0; c < n; c++) {
            vector<int> col;
            col.reserve(n);
            for(int i = 0; i < n; i++)
            col.push_back(grid[i][c]);
            ans += count(begin(grid), end(grid), col);
        }
        return ans;
    }
};

class FoodRatings {
    unordered_map<string, set<pair<int, string>>> cus_itm;
    unordered_map<string, string> fd_cus;
    unordered_map<string, int> fd_rat;

    FoodRatings(vector<string>& fd, vector<string>& cus, vector<int>& rat) {
        int n = fd.size();
        for(int i = 0; i < n; i++) {
            cus_itm[cus[i]].insert(pair(-rat[i], fd[i]));
            fd_cus[fd[i]] = cus[i];
            fd_rat[fd[i]] = rat[i];

        fd_rat[fd] = rat;
 
    }
    
    string highestRated(string cus) {
