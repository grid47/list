string reverseParentheses(string s) {
    stack<int> stk;
    string res;
    
    for(int i = 0; i < s.size(); i++) {
        if(s[i] == '(') 
            stk.push(res.size());
        else if(s[i] == ')') {
            int j = stk.top();
            stk.pop();
            reverse(res.begin() + j, res.end());
        } else res.push_back(s[i]);
    }
    
    return res;
}

int nthUglyNumber(int k, int A, int B, int C) {
    int lo = 1, hi = 2* (int)1e9;
    long a = long(A), b = long(B), c = long(C);
    long ab = a * b / __gcd(a, b);
    long bc = b * c / __gcd(b, c);
    long ac = a * c / __gcd(a, c);
    long abc = a * bc / __gcd(a, bc);
    while (lo < hi ) {
        int mid = lo + (hi - lo)/2;
        int cnt = mid / a + mid / b + mid / c - mid / ab - mid / bc - mid / ac + mid / abc;
        
        if (cnt < k) {
            lo = mid + 1;
        } else hi = mid;
    }
    return lo;
}

    vector<char> ans(n, '-');

    }

}

int equalSubstring(string s, string t, int mxc) {
    vector<int> nums(s.size(), 0);
    for(int i = 0; i < s.size(); i++) {
        nums[i] = abs(s[i] - t[i]);
    }
    int res = 0, j = 0;
    for(int i = 0; i < nums.size(); i++) {
        mxc -= nums[i];
        while(mxc < 0) {
            mxc += nums[j++];
        }
        res = max(res, i - j + 1);
    }
    return res;
}
