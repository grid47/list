string replaceDigits(string s) {

    return s;

int getMinSwaps(string num, int k) {
    string og = num;
    while(k--) {
        next_permutation(num.begin(), num.end());
    }
    return steps(og, num);
}

int steps(string s1, string s2) {
    int sz = s1.size();
    int i = 0, j = 0;
    int res = 0;

    while(i < sz) {
        j = i;
        while(s1[i] != s2[j]) j++;
        while(i < j) {
            swap(s2[j], s2[j - 1]);
            j--;
            res++;
        }
        i++;
    }
    return res;
}

vector<int> memLeak(int mem1, int mem2) {
    int i = 1;
    while(i <= mem1 || i <= mem2) {
        if(mem1 >= mem2) {
            mem1 -= i;
        } else mem2 -= i;
        i++;
    }
    return vector<int>{i, mem1, mem2};
}

int subsetXORSum(vector<int>& nums) {
    int res = 0;
    for (auto i = 1; i < (1 << nums.size()); ++i) {
        int total = 0;
        for (auto j = 0; j < nums.size(); ++j)
            if (i & (1 << j))
                total ^= nums[j];
        res += total;
    }
    return res;
}
