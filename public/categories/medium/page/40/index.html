Leetcode 684: Redundant Connection

You are given a string s containing only the characters '(' and ')'. A string is considered balanced if each '(' has a matching pair of consecutive '))' and if they appear in the correct order. Your task is to return the minimum number of insertions needed to make the string balanced.

int minInsertions(string s) {
    int res = 0;
    
    int n = s.size();
    
    stack<char> stk;
    for(int i = 0; i < n; i++) {
        if(s[i] == '(') {
            stk.push('(');
            continue;
        }
        
        if(i + 1 < n && s[i + 1] == ')') {
            if(!stk.empty()) stk.pop();
            else res++;
            i++;
        } else {
            if(!stk.empty()) {stk.pop(); res++;}
            else res += 2;
        }
        
    }
    
    return res + stk.size() * 2;
}

int maxDistance(vector<int>& pos, int m) {
    sort(pos.begin(), pos.end());
    int n = pos.size();
    int l = 0;
    int r = pos[n - 1]-pos[0];
    while(l < r) {
        int mid = r - (r - l) / 2;
        if(isValid(pos, mid, m)) {
            l = mid;
        } else {
            r = mid - 1;
        }
    }
    return l;
}
bool isValid(vector<int> &nums, int mid, int cnt) {
    int res = 1, cur = nums[0];
    for(int i = 1; i < nums.size(); i ++) {
        if(nums[i] - cur >= mid) {
            res++;
            cur = nums[i];
        }
    }
    return res >= cnt;
}

int findLatestStep(vector<int>& arr, int m) {

    int n = arr.size();
    vector<int> len(n + 2, 0), cnt(n + 1, 0);
    
    int res = -1;
    
    for(int i = 0; i < n; i++) {
        
        int a = arr[i];
        int ll = len[a - 1];
        int rl = len[a + 1];
        len[a] = len[a - ll] = len[a + rl] = ll + rl + 1;
        
        cnt[ll]--;
        cnt[rl]--;
        
        cnt[len[a]]++;
        
        if(cnt[m] > 0) res = i + 1;
        
    }
    
    return res;
}

• If the total is not divisible by 3, return 0.

int numWays(string s) {
    
    long n = s.size();
    int one = 0;
    for(char x: s)
        one += (x == '0')? 0: 1;
    
    int mod = (int) 1e9 + 7;
    
    if(one == 0) return (int)((n - 2) * (n - 1) / 2 % mod);
    
        
    if(one % 3 != 0) return 0;
    
    long long waysOfFirstCut = 0, waysOfSecondCut = 0;
    int net = one / 3, tmp = 0;
    for(int i = 0; i < n; i++) {
        if(s[i] == '1') tmp++;
        if(tmp == net) waysOfFirstCut++;
        else if(tmp == 2 * net) waysOfSecondCut++;
    }
    
    return (int)(waysOfFirstCut *waysOfSecondCut % mod) ;
}
