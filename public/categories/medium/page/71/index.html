int longestNiceSubarray(vector<int>& nums) {
    int res = 1, AND = 0, n = nums.size(), j = 0;
    for(int i = 0; i < n; i++) {
        while(((AND & nums[i]) > 0)) {
            AND ^= nums[j++];
        }
        AND |= nums[i];
        res = max(res, i - j + 1);
    }
    return res;
}
