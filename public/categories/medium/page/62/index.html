Leetcode 973: K Closest Points to Origin

int minimumCost(vector<int>& cost) {
    sort(cost.begin(), cost.end());
    int n = cost.size(), res = 0;
    for(int i = 0; i < n; i++) {
      if(i % 3 == n % 3) continue;
      res += cost[i];
    }
    return res;
}

int numberOfArrays(vector<int>& diff, long lower, long upper) {
    // last - fist of n + 1 seq
    
    long mn = 0, mx = 0, x = 0;

    for(int k : diff) {
        x += k;
        mx = max(mx, x);
        mn = min(mn, x);
    }
    
    int sum = mx - mn;
    if (upper - lower < sum)
        return 0;
    else {
        
        return (int)(upper - lower) - sum + 1;
        
    }
    
}

vector<int> findLonely(vector<int>& nums) {
    unordered_map<int, int> mp;
    for(int num: nums) mp[num]++;

    vector<int> res;
    for(auto [x, cnt]: mp) {
        if(cnt == 1 && !mp.count(x - 1) && !mp.count(x +1))
            res.push_back(x);
    }

    return res;
}

vector<int> sortEvenOdd(vector<int>& nums) {
    int minIndex;
// Here in this nested loop , we are sorting the elements at even indices in non-decreasing order.
    for(int i=0;i<nums.size();i+=2)
    {
        minIndex=i;
        for(int j=i+2;j<nums.size();j+=2)
        {
            if(nums[j]<nums[minIndex])
                minIndex=j;
        }
        swap(nums[i],nums[minIndex]);

        {

                minIndex=j;
        }

    }
    
    return nums;
}
