    long N = 0, i;
    
    for(i = 0; N < k; i++)
        N = isdigit(s[i])? (s[i] - '0') * N: N +1;

        if(isdigit(s[i])) {

            k %= N;

int numRescueBoats(vector<int>& people, int limit) {
    sort(people.begin(), people.end());
    int lo = 0, hi = people.size() - 1;
    int boats = 0;
    while(lo <= hi) {
        
        if(people[lo] + people[hi] <= limit) {
            lo += 1;
            hi -= 1;
        } else hi -= 1;
        
        boats++;
    }
    
    return boats;
}

bool possibleBipartition(int n, vector<vector<int>>& dlk) {
    UF* uf = new UF(n + 1);
    vector<vector<int>> gph(n + 1);
    for(int i = 0; i < dlk.size(); i++) {
        gph[dlk[i][0]].push_back(dlk[i][1]);
        gph[dlk[i][1]].push_back(dlk[i][0]);            
    }

    for(int i = 1; i <= n; i++) {
        for(int x: gph[i]) {
            if(uf->find(x) == uf->find(i)) return false;
            uf->uni(gph[i][0], x);
        }
    }
    return true;
}

bool isMonotonic(vector<int> A) {
    bool inc = true, dec = true;
    for (int i = 1; i < A.size(); ++i)
        inc &= A[i - 1] <= A[i], dec &= A[i - 1] >= A[i];
    return inc || dec;
}

The string "world" does not contain all characters required by `words2`, so it's not a valid answer. The string "goodbye" is also valid as it contains all the characters required by `words2`.

Approach: To solve this problem efficiently, we first calculate the required frequencies for all characters in `words2`. Then, we check each string in `words1` to see if it meets these frequency requirements. If a string contains the necessary characters in the correct amounts, we add it to the result.

vector<string> wordSubsets(vector<string>& words1, vector<string>& words2) {
    vector<int> frq(26, 0);
    for(auto s: words2) {
        vector<int> tmp(26, 0);
        for(char c: s) {
            tmp[c - 'a']++;
            frq[c - 'a'] = max(frq[c - 'a'], tmp[c-'a']);
        }
    }
    vector<string> res;
    for(auto s: words1) {
        vector<int> tmp(26, 0);
        for(char c: s) tmp[c - 'a']++;
        int flag = true;
        for(int i = 0; i < 26; i++) {
            if(tmp[i] < frq[i]) {
                flag = false;
                break;
            }
        }
        if(flag) res.push_back(s);
    }
    return res;
}

int maxSubarraySumCircular(vector<int>& nums) {
    int lmax = nums[0], lmin = nums[0];
    int gmax = nums[0], gmin = nums[0];
    int sum = nums[0];
    for(int i = 1; i < nums.size(); i++) {
        sum += nums[i];
        lmax = max(nums[i], lmax + nums[i]);
        gmax = max(lmax, gmax);
        lmin = min(nums[i], lmin + nums[i]);
        gmin = min(lmin, gmin);
    }
    return gmax > 0? max(gmax, sum - gmin): gmax;
}
