Leetcode 692: Top K Frequent Words

You are given a permutation of integers from 1 to n, where each integer represents a position in a binary string of size n that is initially filled with zeros. At each step, you will flip the bit at the position specified by the current element of the array. You are also given an integer m, and your task is to determine the latest step at which there exists a contiguous substring of exactly m ones in the binary string.

Approach: To solve this problem efficiently, we need to simulate the process of flipping bits at the positions specified by arr and then check for groups of ones at each step.

int findLatestStep(vector<int>& arr, int m) {

    int n = arr.size();
    vector<int> len(n + 2, 0), cnt(n + 1, 0);
    
    int res = -1;
    
    for(int i = 0; i < n; i++) {
        
        int a = arr[i];
        int ll = len[a - 1];
        int rl = len[a + 1];
        len[a] = len[a - ll] = len[a + rl] = ll + rl + 1;
        
        cnt[ll]--;
        cnt[rl]--;
        
        cnt[len[a]]++;
        
        if(cnt[m] > 0) res = i + 1;
        
    }
    
    return res;
}

Leetcode 712: Minimum ASCII Delete Sum for Two Strings

int maximalNetworkRank(int n, vector<vector<int>>& roads) {
    
    vector<int> inward(n, 0);
    map<int, set<int>> mp;
    for(auto it: roads) {
        inward[it[0]]++;
        inward[it[1]]++;
        mp[it[0]].insert(it[1]);
        mp[it[1]].insert(it[0]);
    }
    
    int mx = 0;

    for(int i = 0; i < n; i++) {
        for(int j = 0; j < n; j++) {
            if(i == j) continue;
            mx = max(mx, inward[i] + inward[j] - (mp.count(i) && mp[i].count(j)? 1: 0));
        }
    }
    return mx;
}

class Solution {

public:
int minimumEffortPath(vector<vector<int>>& h) {
    int dir[] = {0, 1, 0, -1, 0};
    int m = h.size(), n = h[0].size();
    vector<vector<int>> sz(m, vector<int>(n, INT_MAX));
    priority_queue<array<int, 3>, vector<array<int, 3>>, cmp> pq;
    pq.push({0, 0, 0});
    while(!pq.empty()) {
        array<int, 3> r = pq.top();
        pq.pop();
        int i = r[0], j = r[1], d = r[2];
        //cout << i << ' ' << j << ' ' << d << endl;
        if(d > sz[i][j]) continue;
        if((i == (m - 1)) && (j == (n - 1))) return d;
        for(int k = 0; k < 4; k++) {
            int ni = i + dir[k], nj = j + dir[k + 1];
            if(min(ni, nj) < 0 || ni >= m || nj >= n) continue;

            int nxt = sz[ni][nj];
            int nd  = max(abs(h[i][j] - h[ni][nj]), d);
            if (nxt > nd) {
                sz[ni][nj] = nd;
                pq.push({ni, nj, nd});
            }
        }
    }
    return 0;
}

int countSubstrings(string s, string t) {
    int res = 0;
    for(int i = 0; i < s.size(); i++)
        res += helper(s, t, i, 0);
    for(int j = 1; j < t.size(); j++)
        res += helper(s, t, 0, j);
    return res;
}

int helper(string s, string t, int i, int j) {
    int res = 0, pre = 0, cur = 0;
    for(int n = s.size(), m = t.size(); i < n && j < m; i++, j++) {
        cur++;
        if(s[i] != t[j]) {
            pre = cur;
            cur = 0;
        }
        res += pre;
    }
    return res;
}
