int longestCommonSubsequence(string t1, string t2) {
    int n1 = t1.size(), n2 = t2.size();
    vector<vector<int>> dp(n1 +1, vector<int>(n2+1, 0));
    for(int i = 0; i < n1; i++)
        for(int j = 0; j < n2; j++)
            if(t1[i] == t2[j])
                dp[i+1][j+1] = dp[i][j] +1;
    else dp[i+1][j+1] = max(dp[i][j+1], dp[i+1][j]);
    
    return dp[n1][n2];
}

string reverseParentheses(string s) {
    stack<int> stk;
    string res;
    
    for(int i = 0; i < s.size(); i++) {
        if(s[i] == '(') 
            stk.push(res.size());
        else if(s[i] == ')') {
            int j = stk.top();
            stk.pop();
            reverse(res.begin() + j, res.end());
        } else res.push_back(s[i]);
    }
    
    return res;
}

int nthUglyNumber(int k, int A, int B, int C) {
    int lo = 1, hi = 2* (int)1e9;
    long a = long(A), b = long(B), c = long(C);
    long ab = a * b / __gcd(a, b);
    long bc = b * c / __gcd(b, c);
    long ac = a * c / __gcd(a, c);
    long abc = a * bc / __gcd(a, bc);
    while (lo < hi ) {
        int mid = lo + (hi - lo)/2;
        int cnt = mid / a + mid / b + mid / c - mid / ab - mid / bc - mid / ac + mid / abc;
        
        if (cnt < k) {
            lo = mid + 1;
        } else hi = mid;
    }
    return lo;
}

int numberOfSubarrays(vector<int>& nums, int k) {
    return atmost(nums, k) - atmost(nums, k - 1);
}

int atmost(vector<int> &nums, int k) {
    int cnt[2] = {};
    int res = 0, j = 0;
    for(int i = 0; i < nums.size(); i++) {
        cnt[nums[i]%2]++;
        while(cnt[1] > k) {
            cnt[nums[j]%2]--;
            j++;
        }
        res += (i - j + 1);
    }
    return res;
}

string minRemoveToMakeValid(string s) {
    vector<int> stk, itk;
    for(int i = 0; i < s.length(); i++) {
        char a = s[i];
             if(a == '(') { stk.push_back(i); }
        else if(a == ')') {
                 if(stk.size()    == 0)    stk.push_back(i);
            else if(s[stk.back()] == '(')  stk.pop_back();
            else stk.push_back(i);
        }
    }
    string res = "";
    set<int> st(stk.begin(), stk.end());
    for(int i = 0; i < s.length(); i++) {
        if(st.count(i)) continue;
        res += s[i];
    }
    return res;        
}
