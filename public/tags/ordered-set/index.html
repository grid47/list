int minOperations(vector<int>& nums, int x) {
    int res = -1;
    long sum = -x;
    for(int y: nums) sum += y;
    int n = nums.size();
    if (sum == 0) return n;
    
    map<int, int> mp;
    mp[0] = -1;
    int s = 0;
    for(int i = 0; i < n; i++) {
        s += nums[i];
        if(mp.count(s - sum)) {
            res = max(res, i - mp[s - sum]);
        }
        mp[s] = i;
    }
    
    return res == -1? res :n - res;        
}

int minLengthAfterRemovals(vector<int>& nums) {
    int n = nums.size();
    unordered_map<int, int> mp;
    for(int i : nums){
        mp[i]++;
    }
    int maxi = 0;
    for(auto it : mp){
        maxi = max(maxi, it.second);
    }
    if(maxi <= n/2){
        if(n%2){

            return 0;
        }
    }
