string kthDistinct(vector<string>& arr, int k) {
    unordered_map<string, int> mp;
    for (auto &s : arr)
        ++mp[s];
    for (auto &s : arr)
        if (mp[s] == 1 && --k == 0)
            return s;
    return "";
}

int numberOfArrays(vector<int>& diff, long lower, long upper) {
    // last - fist of n + 1 seq
    
    long mn = 0, mx = 0, x = 0;

    for(int k : diff) {
        x += k;
        mx = max(mx, x);
        mn = min(mn, x);
    }
    
    int sum = mx - mn;
    if (upper - lower < sum)
        return 0;
    else {
        
        return (int)(upper - lower) - sum + 1;
        
    }
    
}

int subarrayGCD(vector<int>& nums, int k) {        
    int cnt = 0, n = nums.size();
    
    for(int i = 0; i < n; i++) {
        for(int j = i; j < n && nums[j] % k == 0; j++) {
            nums[i] = __gcd(nums[i], nums[j]);
            cnt += (nums[i] == k);
        }
    }
    return cnt;
}
