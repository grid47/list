    while(i < j && a[i] == b[j])
    i++, j--;

int minOperations(vector<int>& nums, int x) {
    int res = -1;
    long sum = -x;
    for(int y: nums) sum += y;
    int n = nums.size();
    if (sum == 0) return n;
    
    map<int, int> mp;
    mp[0] = -1;
    int s = 0;
    for(int i = 0; i < n; i++) {
        s += nums[i];
        if(mp.count(s - sum)) {
            res = max(res, i - mp[s - sum]);
        }
        mp[s] = i;
    }
    
    return res == -1? res :n - res;        
}

int minAbsoluteSumDiff(vector<int>& nums1, vector<int>& nums2) {

    int n = nums1.size();
    set<int> sel(begin(nums1), end(nums1));

    long gain = 0, res = 0;
    for(int i = 0; i < n; i++) {
        long og = abs(nums1[i] - nums2[i]);
        res += og;
        if (gain < og) {
            auto num = sel.lower_bound(nums2[i]);
            if (num !=   end(sel)) gain = max(gain, og - abs(*num       - nums2[i]));
            if (num != begin(sel)) gain = max(gain, og - abs(*prev(num) - nums2[i]));
        }
    }

    return (res - gain) % 1000000007;
}

Approach: The solution involves sorting the array and performing operations to reduce the largest element until all elements are equal.
Observations:
• Sorting the array ensures that we always have easy access to the largest and next largest values.
• By iterating from the largest values downward, we can minimize the number of operations needed to make all elements the same.
Steps:
• Sort the nums array in ascending order.
• Iterate through the array in reverse order and perform operations to reduce the largest element to the next smaller element.
• Count the operations until all elements in the array are equal.
Empty Inputs:
• The input array cannot be empty, as the problem guarantees at least one element.
Large Inputs:
• For larger inputs (e.g., nums.length = 50,000), the solution should handle them efficiently with O(n log n) time complexity.
Special Values:
• The array could contain repeated values or be already equal, in which case no operations would be required.
Constraints:
• Ensure that the solution works within the time and space limits for larger arrays.
int reductionOperations(vector<int>& nums) {

    int ops = 0, n = nums.size();
    sort(nums.begin(), nums.end());
    for(int i = n - 2; i >= 0; i--) {
        
        if(nums[i] == nums[i + 1]) continue;
        ops += (n - 1 - i);
        
    }
    
    return ops;
}
