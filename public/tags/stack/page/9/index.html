public:

        bwd.push_back(homepage);

    }
    
    void visit(string url) {
        bwd.push_back(url);
        fwd.resize(0);
    }
    
    string back(int steps) {
        int x = bwd.size();
        while(steps-- > 0 && bwd.size() > 1) {
            string p = bwd.back();
            fwd.push_back(p);
            bwd.pop_back();

Code:

class Solution {
public:
    int numSubmat(vector<vector<int>>& mat) {
        int m = mat.size();
        int n = mat[0].size();
        int cnt = 0;
        for(int i = 0; i < m; i++)
        for(int j = 0; j < n; j++)
            cnt += helper(mat, i, j);        
        return cnt;
    }

    int helper(vector<vector<int>>& mat, int a, int b) {
        int m = mat.size(), n = mat[0].size();
        int cnt = 0, bnd = n;
        for(int i = a; i < m; i++)
        for(int j = b; j < bnd; j++) {
            if(mat[i][j]) cnt +=1;
            else bnd = j;
        }
        return cnt;
    }
};

Leetcode 1653: Minimum Deletions to Make String Balanced

Code:

class Solution {
public:
    int minimumDeletions(string s) {
        int n = s.size(), bcnt = 0;
        vector<int> dp(n + 1, 0);
        for(int i = 0; i < n; i++) {
            char a = s[i];
            if(a == 'a') {                
                dp[i + 1] = min(dp[i] + 1, bcnt);
            } else {
                bcnt++;
                dp[i + 1] = dp[i];
            }
        }
        return dp[n];
    }
};

Problem Statement

The problem involves determining the minimum number of deletions required to ensure that all characters in a given string s are of the same type, specifically the character ‘a’. In the string, each character is either ‘a’ or ‘b’. The goal is to compute the minimum deletions necessary to convert the string into one consisting entirely of ‘a’s.

Leetcode 1673: Find the Most Competitive Subsequence

Code:

class Solution {
public:
    vector<int> mostCompetitive(vector<int>& nums, int k) {

        vector<int> stk;
        for(int i = 0; i < nums.size(); i++) {

            while (!stk.empty()          &&
                 stk.back() > nums[i] &&
                (stk.size() + nums.size() - (i + 1)) >= k )
                stk.pop_back();

            if(stk.size() < k)
                stk.push_back(nums[i]);

        }
        
        return stk;
        
    }
};

Problem Statement

The problem is to determine the most competitive subsequence of a given length ( k ) from an array of integers ( nums ). A subsequence is considered more competitive if it appears in the order of elements in the original array while being lexicographically smaller than any other subsequence of the same length. The goal is to return a subsequence of size ( k ) that meets these criteria.
