long long zeroFilledSubarray(vector<int>& nums) {
    long long res = 0;
    for(int i = 0, j = 0; i < nums.size(); i++) {
        if(nums[i] != 0)
        j = i +1;
        res += i + 1 - j;

}

vector<int> leftRightDifference(vector<int>& nums) {
    int n = nums.size();
    vector<int> sum(n + 1, 0), ans(n, 0);
    for(int i = 0; i < n; i++)
        sum[i + 1] = nums[i] + sum[i];
    
    int net = sum[n];
    for(int i = 0; i < n; i++) {
        ans[i] = abs(net - sum[i + 1] - sum[i]);
    }
    return ans;
}

int kItemsWithMaximumSum(int one, int zero, int neg, int k) {
    int sum = 0;
    sum += min(one, k);
    k -= min(one, k);
    if(k > 0) {
        k -= min(zero, k);
    }
    if(k > 0) {
        sum -= min(neg, k);
        k -= min(neg, k);
    }
    return sum;
}
