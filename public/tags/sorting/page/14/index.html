bool isAcronym(vector<string>& words, string s) {
    string res = "";
    for(string w: words)
        res += w[0];
    return s == res;
}

int furthestDistanceFromOrigin(string moves) {
    int n=moves.length();
    
    int l=0, r=0;
    for(int i=0;i<n;i++){
        if(moves[i]=='L' || moves[i]=='_'){
            l--;
        }else{
            l++;
        }
        
        if(moves[i]=='R' || moves[i]=='_'){
            r++;
        }else{
            r--;
        }
    }
    
    if(l<0) l *= -1;
    if(r<0) r *= -1;
		
    return max(l,r);
}

int minLengthAfterRemovals(vector<int>& nums) {
    int n = nums.size();
    unordered_map<int, int> mp;
    for(int i : nums){
        mp[i]++;
    }
    int maxi = 0;
    for(auto it : mp){
        maxi = max(maxi, it.second);
    }
    if(maxi <= n/2){
        if(n%2){
            return 1;
        }
        else{
            return 0;
        }
    }
    return 2*maxi - n;
