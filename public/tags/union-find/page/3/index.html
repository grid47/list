Leetcode 1061: Lexicographically Smallest Equivalent String

• Each time we encounter a '|', we flip the sign to track whether we are inside a pair of bars, and count the asterisks outside.

• 2. Use a sign variable to track whether we are inside a pair of vertical bars.

int countAsterisks(string s) {
    int res = 0, sign = 1;
    for (char& c : s)
        if ((sign ^= c == '|') && c == '*')
            res++;
    return res;
}

Approach: The approach is to iterate through each number in 'nums' and for each divisor, count how many numbers in 'nums' are divisible by that divisor. Then, we return the divisor with the highest count.

int maxDivScore(vector<int>& nums, vector<int>& div) {
    int n = div.size();
    int m = nums.size();
    vector<int> cnt(n, 0);
    int mx = 0, idx = *min_element(div.begin(), div.end());
    map<int, vector<int>> mp;
    for(int i = 0; i < m; i++) {
        if(mp.count(nums[i])) {
            for(int x: mp[nums[i]]) {
                cnt[x]++;
                if(cnt[x] > mx || (cnt[x] == mx && div[x] < div[idx])) {
                    mx = cnt[x];
                    idx = x;
                }
            }
        } else {
            for(int j = 0; j < n; j++) {
                if(nums[i] % div[j] == 0) {
                    mp[nums[i]].push_back(j);
                    cnt[j]++;
                    if(cnt[j] > mx || (cnt[j] == mx && div[j] < div[idx])) {
                        mx = cnt[j];
                        idx = j;
                    }                        
                }
            }
        }
    }
    return mx == 0? idx:div[idx];
}

int minIncrements(int n, vector<int>& A) {
    int res = 0;
    for (int i = n / 2 - 1; i >= 0; --i) {
        int l = i * 2 + 1, r = i * 2 + 2;
        res += abs(A[l] - A[r]);
        A[i] += max(A[l], A[r]);
    }
    return res;
}
