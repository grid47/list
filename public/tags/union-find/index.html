int kthSmallest(TreeNode* root, int k) {
    stack<TreeNode*> stk;
    // stk.push(root);
    TreeNode* node= root;
    while(!stk.empty() || node) {
        if(node) {
            stk.push(node);
            node = node->left;
        } else {
            node = stk.top();
            stk.pop();
            k--;
            if(k == 0) return node->val;
            node = node->right;
        }
    }
    return NULL;
}

int eraseOverlapIntervals(vector<vector<int>>& ivl) {
    sort(ivl.begin(), ivl.end());
    int ans = 0;

    int prv = 0;
    for(int cur = 1; cur < n; cur++) {
        // [1, 5]. [3, 6]

        if(ivl[cur][0] < ivl[prv][1]) {
            ans++;
            if(ivl[cur][1] <= ivl[prv][1])
                prv = cur;
        } else {
            prv = cur;
        }
    }
    return ans;
}

int minInsertions(string s) {
    int res = 0;
    
    int n = s.size();
    
    stack<char> stk;
    for(int i = 0; i < n; i++) {
        if(s[i] == '(') {
            stk.push('(');
            continue;
        }
        
        if(i + 1 < n && s[i + 1] == ')') {
            if(!stk.empty()) stk.pop();
            else res++;
            i++;
        } else {
            if(!stk.empty()) {stk.pop(); res++;}
            else res += 2;
        }
        
    }
    
    return res + stk.size() * 2;
}
