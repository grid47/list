Leetcode 692: Top K Frequent Words

You are given a permutation of integers from 1 to n, where each integer represents a position in a binary string of size n that is initially filled with zeros. At each step, you will flip the bit at the position specified by the current element of the array. You are also given an integer m, and your task is to determine the latest step at which there exists a contiguous substring of exactly m ones in the binary string.

Approach: To solve this problem efficiently, we need to simulate the process of flipping bits at the positions specified by arr and then check for groups of ones at each step.

Steps:
• Initialize an array to keep track of the lengths of contiguous 1's around each flipped bit.
• For each step, flip the bit at the index specified by arr.
• After flipping, update the group lengths, check if a group of length m exists, and track the latest step when this happens.

int findLatestStep(vector<int>& arr, int m) {

    int n = arr.size();
    vector<int> len(n + 2, 0), cnt(n + 1, 0);
    
    int res = -1;
    
    for(int i = 0; i < n; i++) {
        
        int a = arr[i];
        int ll = len[a - 1];
        int rl = len[a + 1];
        len[a] = len[a - ll] = len[a + rl] = ll + rl + 1;
        
        cnt[ll]--;
        cnt[rl]--;
        
        cnt[len[a]]++;
        
        if(cnt[m] > 0) res = i + 1;
        
    }
    
    return res;
}

static int minOperations(vector<string>& logs) {
    int level=0;
    for(auto& dir: logs){
        if (dir=="../") 
            level-=(level>0);
        else if (dir!="./")
            level++;
    //    cout<<level<<endl;
    }
    return level;
}

class Solution {

public:
int minimumEffortPath(vector<vector<int>>& h) {
    int dir[] = {0, 1, 0, -1, 0};
    int m = h.size(), n = h[0].size();
    vector<vector<int>> sz(m, vector<int>(n, INT_MAX));
    priority_queue<array<int, 3>, vector<array<int, 3>>, cmp> pq;
    pq.push({0, 0, 0});
    while(!pq.empty()) {
        array<int, 3> r = pq.top();
        pq.pop();
        int i = r[0], j = r[1], d = r[2];
        //cout << i << ' ' << j << ' ' << d << endl;
        if(d > sz[i][j]) continue;
        if((i == (m - 1)) && (j == (n - 1))) return d;
        for(int k = 0; k < 4; k++) {
            int ni = i + dir[k], nj = j + dir[k + 1];
            if(min(ni, nj) < 0 || ni >= m || nj >= n) continue;

            int nxt = sz[ni][nj];
            int nd  = max(abs(h[i][j] - h[ni][nj]), d);
            if (nxt > nd) {
                sz[ni][nj] = nd;
                pq.push({ni, nj, nd});
            }
        }
    }
    return 0;
}

    sort(pts.begin(), pts.end());
    int res =0;

int minMoves(vector<int>& num, int lmt) {
    int n = num.size();
    vector<int> dp(2*lmt +2,0);
    for(int i = 0; i < n/2; i++) {
        int a = num[i];
        int b = num[n-1-i];
        dp[2] += 2;
        dp[min(a,b)+1] -= 1;

Leetcode 752: Open the Lock

A restaurant with a single chef serves customers. Each customer arrives at a specific time and waits for the chef to prepare their order. The chef can only serve one customer at a time and prepares orders in the order they were received. The goal is to calculate the average waiting time of all customers. The waiting time of a customer is the time between their arrival and when they receive their order.

double averageWaitingTime(vector<vector<int>>& cust) {
    
    int cur = cust[0][0];
    
    double n = cust.size();
    double sum = 0;
    
    for(int i = 0; i < n; i++) {
        
        if(cur > cust[i][0])
        cur += cust[i][1];
        else
        cur = cust[i][0] + cust[i][1];
        
        sum += cur - cust[i][0];
        
    }
    
    return sum / n;
}
