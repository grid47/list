int countTriples(int n) {
vector<bool> squares(n * n + 1);
for (int i = 1; i <= n; ++i)
    squares[i * i] = true;
int res = 0;
for (int i = 1; i <= n; ++i)
    for (int j = i; i * i + j * j <= n * n; ++j)
        res += squares[i * i + j * j] * 2;
return res;
}

Leetcode 889: Construct Binary Tree from Preorder and Postorder Traversal

Approach
Steps
Complexity

Observations:

int countPaths(int n, vector<vector<int>>& rds) {
    vector<vector<pll>> gph(n);
    for(vector<int> &e : rds) {
        int u = e[0], v = e[1], time = e[2];
        gph[u].push_back({v, time});
        gph[v].push_back({u, time});
        
    }
    
    vector<ll> ways(n, 0), dst(n, LLONG_MAX);
    ways[0] = 1;
     dst[0] = 0;
    
    priority_queue<pll, vector<pll>, greater<>> pq;
    
    pq.push({0, 0});
    int ans = 0;
    
    while(!pq.empty()) {
        
      auto [d, u] = pq.top();
      pq.pop();
        
   for(pll nb : gph[u]) {
          int nxt = nb.first, t = nb.second;
   if(dst[nxt] > d + t) {
       
       dst[nxt] = d + t;
       ways[nxt] = ways[u];
       pq.push({d + t, nxt});
   } else if (dst[nxt] == d + t) {
       ways[nxt] = (ways[nxt] + ways[u]) % 1000000007;
   }
   
   }
        
    }
    
    return ways[n - 1];
}

Approach: To solve this problem, we need to simulate the game and determine who will win given the sequence of stones and the optimal play strategy.

bool stoneGameIX(vector<int>& stones) {
    vector<int> m(3, 0);
    for(int a: stones)
    m[a % 3]++;
    if(min(m[2], m[1]) == 0)
    return max(m[1], m[2]) > 2 && m[0] % 2 >0;
    return abs(m[1] - m[2]) > 2 || m[0] % 2 == 0;
}
