int numTilings(int n) {
    int mod = 1e9 + 7;
    vector<long long> v(10001, 0);
    v[1] = 1;
    v[2] = 2;
    v[3] = 5;
    if(n <= 3) return v[n];
    for(int i = 4; i <= n; i++) {
        v[i]  = (2 * v[i-1] + v[i-3]);
        v[i] %= mod;
    }
    return v[n];
}

vector<string> ambiguousCoordinates(string S) {
    int n = S.size();
    vector<string> res;
    for (int i = 1; i < n - 2; ++i) {
        vector<string> A = f(S.substr(1, i)), B = f(S.substr(i + 1, n - 2 - i));
        for (auto & a : A) for (auto & b : B) res.push_back("(" + a + ", " + b + ")");
    }
    return res;
}
vector<string> f(string S) {
    int n = S.size();
    if (n == 0 || (n > 1 && S[0] == '0' && S[n - 1] == '0')) return {};
    if (n > 1 && S[0] == '0') return {"0." + S.substr(1)};
    if (n == 1 || S[n - 1] == '0') return {S};
    vector<string> res = {S};
    for (int i = 1; i < n; ++i) res.push_back(S.substr(0, i) + '.' + S.substr(i));
    return res;
}

bool isNStraightHand(vector<int>& hand, int groupSize) {
    map<int, int> mp;
    for(int x: hand) mp[x]++;
    
    for(auto it: mp) {
        if(mp[it.first] > 0)
        for(int i = 1; i < groupSize; i++){
            mp[it.first + i] -= mp[it.first];
            if(mp[it.first + i] < 0)
                return false;
        }
        mp[it.first] = 0;
    }
    return true;
}
