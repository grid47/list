int mctFromLeafValues(vector<int>& arr) {
    
    int res = 0;
    
    vector<int> stk = { INT_MAX };
    for(int a : arr) {
        while(stk.back() <= a) {
            int mid = stk.back();
            stk.pop_back();
            res += mid * min(stk.back(), a);
        }
        stk.push_back(a);
    }
    
    for(int i = 2; i < stk.size(); i++) {
        res += stk[i] * stk[i - 1];
    }
    
    return res;
}

int equalSubstring(string s, string t, int mxc) {
    vector<int> nums(s.size(), 0);
    for(int i = 0; i < s.size(); i++) {
        nums[i] = abs(s[i] - t[i]);
    }
    int res = 0, j = 0;
    for(int i = 0; i < nums.size(); i++) {
        mxc -= nums[i];
        while(mxc < 0) {
            mxc += nums[j++];
        }
        res = max(res, i - j + 1);
    }
    return res;
}

Leetcode 553: Optimal Division

vector<int> circularPermutation(int n, int start) {
    vector<int> res;
    for(int i = 0; i < 1 << n; i++)
        res.push_back(start ^ i ^ i >> 1);
    return res;
}

int deepestLeavesSum(TreeNode* root) {
    if(root == NULL) return 0;
    queue<TreeNode*> q;
    q.push(root);
    int sum = 0;
    while(!q.empty()) {
        int sz = q.size();
        sum = 0;
        while(sz--) {
            sum += q.front()->val;
            TreeNode* x = q.front();
            q.pop();
            if(x->left) q.push(x->left);
            if(x->right) q.push(x->right);
        }
    }
    return sum;
}

int minFlips(int a, int b, int c) {
    // (a|b) has zero and c has 1 required change is one

        }
    }

}

int longestSubarray(vector<int>& nums, int limit) {
    multiset<int> ms;
    int res = 0, j = 0;
    for(int i = 0; i < nums.size(); i++) {
        ms.insert(nums[i]);
        while(!ms.empty() && *ms.rbegin() - *ms.begin() > limit) {
            ms.erase(ms.find(nums[j++]));
        }
        res = max(res, i - j + 1);
    }
    return res;
}
