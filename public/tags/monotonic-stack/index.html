    UF* uf = new UF(n + 1);
    vector<vector<int>> gph(n + 1);
    for(int i = 0; i < dlk.size(); i++) {
        gph[dlk[i][0]].push_back(dlk[i][1]);
        gph[dlk[i][1]].push_back(dlk[i][0]);            
    }

    for(int i = 1; i <= n; i++) {
        for(int x: gph[i]) {
            if(uf->find(x) == uf->find(i)) return false;

        }
    }
    return true;
}

    int m = mat.size(), n = mat[0].size();
    vector<vector<int>> sum(m + 1, vector<int>(n + 1, 0));

    for(int i = 1; i < m + 1; i++)
    for(int j = 1; j < n + 1; j++)
    sum[i][j] = mat[i - 1][j - 1] + (sum[i - 1][j] + sum[i][j - 1] - sum[i - 1][j - 1]);
    
    vector<vector<int>> res(m, vector<int>(n));

    for(int i = 1; i < m + 1; i++)
    for(int j = 1; j < n + 1; j++) {

        int r1 = min(i + k, m);
        int c1 = min(j + k, n);
        int r2 = max(i - k, 1);
        int c2 = max(j - k, 1);

        res[i - 1][j - 1] = sum[r1][c1] - sum[r1][c2 - 1] - sum[r2 - 1][c1] + sum[r2 - 1][c2 - 1];

    }
    
    return res;
}
