void bt(vector<vector<int>> &ans, vector<int> &tmp, vector<int> &nums, int idx, map<int, bool> &mp) {
    if(idx == nums.size()) {
        ans.push_back(tmp);
        return;
    }
    
    for(int i = 0; i < nums.size(); i++) {
        if(mp.count(nums[i])) continue;
        mp[nums[i]] = true;
        tmp.push_back(nums[i]);
        bt(ans, tmp, nums, idx + 1, mp);
        tmp.pop_back();
        mp.erase(nums[i]);
    }
}

void f(vector<int>&nums,vector<vector<int>>&ans,int ind){
    if(ind==nums.size()){
        ans.push_back(nums);
        return;
    }
    unordered_map<int,int>mp;
    for(int i=ind;i<nums.size();i++){
        if(mp.find(nums[i])!=mp.end()){
            continue;
        }else{
            mp[nums[i]]++;
            swap(nums[i],nums[ind]);
            f(nums,ans,ind+1);
            swap(nums[i],nums[ind]);
        }
    }
}

void rotate(vector<vector<int>>& mtx) {
    int i = 0, j = mtx.size() - 1;
    while(i <= j)
        swap(mtx[i++], mtx[j--]);
    
    for(int i = 0; i < mtx.size(); i++)
    for(int j = i+1; j< mtx.size(); j++)
        swap(mtx[i][j], mtx[j][i]);
}

vector<vector<int>> levelOrderBottom(TreeNode* root) {

    vector<vector<int>> ans;        
    queue<TreeNode*> q;
    if(root == NULL) return ans;
    
    q.push(root);        
    stack<vector<int>> stk;
            
    while(!q.empty())
    {
        vector<int> ans;
        int sz = q.size();

        while(sz--) {
            TreeNode* tmp = q.front();
            q.pop();
            ans.push_back(tmp->val);

            if(tmp->left)   q.push(tmp->left);
            if(tmp->right)  q.push(tmp->right);                
        }
        stk.push(ans);
    }

    while(!stk.empty())
    {
        ans.push_back(stk.top());
        stk.pop();
    }

    return ans;      
}
