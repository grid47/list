int minDeletion(vector<int>& nums) {
    int ans = 0;
    for(int i = 0; i < nums.size() - 1; i++)
        if(nums[i] == nums[i + 1] && (i - ans) % 2 == 0) ans++;
    
    return ans + (nums.size() - ans) % 2;
}

int maxJump(vector<int>& A) {
    int res = A[1] - A[0];
    int n = A.size();
    for(int i = 2; i < n; i++)
        res = max(res, A[i] - A[i - 2]);
    return res;
}

vector<int> vowelStrings(vector<string>& words, vector<vector<int>>& q) {
    int n = words.size();
    vector<int> cnt(n, 0), ans(q.size(), 0);
    set<char> ch = {'a', 'e', 'i', 'o', 'u'};
    for(int i = 0; i < n; i++) {
        if(ch.count(words[i][0]) && ch.count(words[i][words[i].size() -1]))
            cnt[i] = 1;
        if(i > 0) cnt[i] += cnt[i - 1];
