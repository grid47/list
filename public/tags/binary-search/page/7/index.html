Leetcode 826: Most Profit Assigning Work

string replaceDigits(string s) {

bool mergeTriplets(vector<vector<int>>& tri, vector<int>& tgt) {
    vector<int> res(3, 0);
    int n = tri.size();

    for(auto s: tri) {
        if(s[0] <= tgt[0] && s[1] <= tgt[1] && s[2] <= tgt[2]) {
            res = { max(res[0], s[0]), max(res[1], s[1]), max(res[2], s[2]) };

int getLucky(string s, int k) {
    string tmp;
    for (char c : s) tmp += to_string(c - 'a' + 1);
    int n = 0;
    for (char c : tmp) n += c - '0';
    for (int i = 1; i < k; ++i) {
        int next = 0;
        while (n) {
            next += n % 10;
            n /= 10;
        }
        n = next;
    }
    return n;
}

int maximumDifference(vector<int>& nums) {
    int mn = nums[0], res = -1;
    for (int i = 1; i < nums.size(); ++i) {
        res = max(res, nums[i] - mn);
        mn = min(mn, nums[i]);
    }
    return res == 0 ? -1 : res;
}
