int findSmallestInteger(vector<int>& nums, int value) {
    map<int, int> mp;
    int n = nums.size();
    for(int i = 0; i < n; i++) {
        mp[(nums[i] < 0? (nums[i] % value + value)%value: nums[i]%value)]++;
    }
    int idx = 0, mn = INT_MAX;
    for(int i = 0; i < value; i++) {
        if(mp[i] < mn) {
            idx = i;
            mn = mp[i];
        }
    }
    return mn * value + idx;
}

For each node i, its left child is 2*i and its right child is 2*i + 1. You are allowed to increment the cost of any node by 1 any number of times. int countSeniors(vector<string>& details) {
    int count = 0;
    for(auto i : details) {
        if(i[11] - '0' > 6) count++;
        else if (i[11] - '0' == 6 && i[12] - '0' > 0) count++;
    }
    return count;
}

int distanceTraveled(int mt, int at) {        
    int net = 0;
    
    while(mt > 0) {
        int red = min(5, mt);
        mt -= red;
        net += red * 10;
        if(red == 5 && at > 0) {
            mt += 1;
            at--;
        }
    }
    
    return net;
}

int furthestDistanceFromOrigin(string moves) {
    int n=moves.length();
    
    int l=0, r=0;
    for(int i=0;i<n;i++){
        if(moves[i]=='L' || moves[i]=='_'){
            l--;
        }else{
            l++;
        }
        
        if(moves[i]=='R' || moves[i]=='_'){
            r++;
        }else{
            r--;
        }
    }
    
    if(l<0) l *= -1;
    if(r<0) r *= -1;
		
    return max(l,r);
}
