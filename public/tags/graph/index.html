int fourSumCount(vector<int>& A, vector<int>& B, vector<int>& C, vector<int>& D) {
    int res = 0;
    unordered_map<int, int> mp;
    for(int a : A)
    for(int b : B)
    mp[a+b]++;
    for(int c : C)
    for(int d : D)
    if(mp.count(-c-d)) res+= mp[-c-d];
    return res;
}

class Solution {

public:
string decodeAtIndex(string s, int k) {
    
    long N = 0, i;
    
    for(i = 0; N < k; i++)
        N = isdigit(s[i])? (s[i] - '0') * N: N +1;
    
    while(i--) {
        if(isdigit(s[i])) {
            N /= (s[i] - '0');
            k %= N;
        }
        else if (k % N-- == 0) {
            return string(1, s[i]);
        }
    }
    
    return "hello yasir";
}
