Leetcode 229: Majority Element II

Given a string s, return the length of the longest palindromic subsequence in s. A subsequence is a sequence derived by deleting some or no elements without changing the order of the remaining elements.

Approach
Steps
Complexity

Steps:

• 2. For each pair of indices `i` and `j`, check if the characters match.
• 3. If the characters match, consider the longest subsequence between `i+1` and `j-1`, and add 2.
• 4. If the characters don't match, take the maximum value between `dp(i + 1, j)` and `dp(i, j - 1)`.
Goal: The string `s` will be between 1 and 1000 characters long, and will only contain lowercase letters.

Approach: The approach uses dynamic programming to calculate the longest palindromic subsequence.

• 3. The value of `dp[0][n-1]` will contain the length of the longest palindromic subsequence of the entire string `s`.

string str;
vector<vector<int>> memo;
int dp(int i, int j) {
    if(i == j) return 1;
    if(i == j - 1) return str[i] == str[j]? 2: 1;
    
    if(memo[i][j] != -1) return memo[i][j];
    
    int ans = 0;
    if(str[i] == str[j]) {
        ans = 2 + dp(i + 1, j - 1);
    } else {
        ans = max(dp(i + 1, j), dp(i, j - 1));
    }
    return memo[i][j] = ans;
}

int longestPalindromeSubseq(string s) {
    str = s;
    int n = s.size();
    memo.resize(n, vector<int>(n, -1));
    return dp(0, n - 1);
}

int numSubarrayBoundedMax(vector<int>& nums, int left, int right) {
    
    int n = nums.size();
    
    int ans = 0, dp = 0, prv = -1;
    for(int i = 0; i < n; i++) {
        
        if(nums[i] < left)
            ans += dp;
            
        if(nums[i] > right) {
            dp = 0;
            prv = i;
        }
            
        if(nums[i] >= left && nums[i] <= right) {
            dp = i - prv;
            ans += dp;
        }
            
    }
    
    return ans;
}

Leetcode 383: Ransom Note

int mirrorReflection(int p, int q) {
    while( p % 2 == 0 && q % 2 == 0) p >>= 1, q >>= 1;
    return 1 - p%2 + q%2;
}

    if(!node) return;
    if(node->left) {
        mp[node->left] = node;
        parents(node->left);
    }

    if(node->right) {
        mp[node->right] = node;
        parents(node->right);

    if(!root) return {};
    parents(root);

    dfs(target, k);
    return ans;
}

void dfs(TreeNode* node, int k) {
    if(st.count(node)) return;
    st.insert(node);

    if(k == 0) {
        ans.push_back(node->val);
        return;
    }

    if(node->left) dfs(node->left, k - 1);
    if(node->right) dfs(node->right, k - 1);
    if(mp[node]) dfs(mp[node], k - 1);
}

class Solution {
int mv;
public:
int distributeCoins(TreeNode* root) {
    mv = 0;
    move(root, mv);
    return mv;
}

int move(TreeNode* r, int & mv) {
    if(r == nullptr) return 0;
    int left = move(r->left, mv);
    int right = move(r->right, mv);
    mv += abs(left) + abs(right);
    return r->val + left + right - 1;
}
