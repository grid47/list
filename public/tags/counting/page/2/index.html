bool canConstruct(string s, int k) {
    bitset<26> odd;
    for(char c: s)
    odd.flip(c - 'a');
    return odd.count() <= k && k <= s.size();
}

Leetcode 692: Top K Frequent Words

At each step, you will flip the bit at the position specified by the current element of the array. You are also given an integer m, and your task is to determine the latest step at which there exists a contiguous substring of exactly m ones in the binary string.

Complexity

Goal: The number of elements in arr will be between 1 and 100,000, and the value of m will be between 1 and n.

Approach: To solve this problem efficiently, we need to simulate the process of flipping bits at the positions specified by arr and then check for groups of ones at each step.

• For each step, flip the bit at the index specified by arr.
• After flipping, update the group lengths, check if a group of length m exists, and track the latest step when this happens.
Empty Inputs:

int findLatestStep(vector<int>& arr, int m) {

    int n = arr.size();
    vector<int> len(n + 2, 0), cnt(n + 1, 0);
    
    int res = -1;
    
    for(int i = 0; i < n; i++) {
        
        int a = arr[i];
        int ll = len[a - 1];
        int rl = len[a + 1];
        len[a] = len[a - ll] = len[a + rl] = ll + rl + 1;
        
        cnt[ll]--;
        cnt[rl]--;
        
        cnt[len[a]]++;
        
        if(cnt[m] > 0) res = i + 1;
        
    }
    
    return res;
}

int largestAltitude(vector<int>& gain) {
    int mx = 0;
    int cur = 0;
    for(int x: gain) {
        cur += x;
        mx = max(mx, cur);
    }
    return mx;
}

    int count =0;
    string ans="";
    for(auto x:s)

int canBeTypedWords(string text, string brokenLetters) {
    bool broken[26] = {};
    for (auto ch : brokenLetters)
        broken[ch - 'a'] = true;
    int res = 0, cnt = 0;
    for (auto ch : text)
        if (ch == ' ') {
            res += cnt == 0;
            cnt = 0;
        }
        else
            cnt += broken[ch - 'a'];
    return res + (cnt == 0);
}

class Solution {

public:
static bool comp(vector<int> &a, vector<int> &b) {
    if(a[0] == b[0]) {
        return a[1] > b[1];
    } else return a[0] < b[0];
}

int numberOfWeakCharacters(vector<vector<int>>& prpt) {
    sort(prpt.begin(), prpt.end(), comp);
    
    int mn = INT_MIN;
    int ans = 0;
    
    for(int i = prpt.size() - 1; i >= 0; i--) {
        if(prpt[i][1] < mn) ans++;
        mn = max(mn, prpt[i][1]);
    }
    return ans;
}
