Leetcode 70: Climbing Stairs

Given the root of a binary tree, your task is to return the level order traversal of its nodes’ values. This means you should traverse the tree level by level, from left to right at each level.

vector<vector<int>> levelOrder(TreeNode* root) {
    vector<vector<int>> ans;
    if(!root) return ans;
    queue<TreeNode*> q;
    q.push(root);
    while(!q.empty()) {
        int sz = q.size();
        vector<int> res;
        while(sz--) {
            TreeNode* tmp = q.front();
            res.push_back(tmp->val);
            q.pop();
            if(tmp->left) q.push(tmp->left);
            if(tmp->right) q.push(tmp->right);
        }
        ans.push_back(res);
    }
    return ans;
}

int numSquares(int n) {
    vector<long> cnt(n + 1, INT_MAX);

    cnt[0] = 0;

    for(int i = 1; i <= n; i++)
    for(int j = 1; j * j <= i; j++)
        cnt[i] = min(cnt[i], cnt[i - j * j] + 1);

    return cnt[n];

}

Leetcode 638: Shopping Offers

int longestSubarray(vector<int>& nums, int limit) {
    multiset<int> ms;
    int res = 0, j = 0;
    for(int i = 0; i < nums.size(); i++) {
        ms.insert(nums[i]);
        while(!ms.empty() && *ms.rbegin() - *ms.begin() > limit) {
            ms.erase(ms.find(nums[j++]));
        }
        res = max(res, i - j + 1);
    }
    return res;
}
