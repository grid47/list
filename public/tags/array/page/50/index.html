int distanceTraveled(int mt, int at) {        
    int net = 0;
    
    while(mt > 0) {
        int red = min(5, mt);
        mt -= red;
        net += red * 10;
        if(red == 5 && at > 0) {
            mt += 1;
            at--;
        }
    }
    
    return net;
}

int rec(int level , int first , int last , vector <string>& arr){

        return 0 ;

vector<long long> countBlackBlocks(int m, int n, vector<vector<int>>& coordinates) {
    unordered_map<long long, int> cnt;
    vector<long long> res = {(n - 1L) * (m - 1), 0, 0, 0, 0};
    for (auto& c: coordinates)
        for (int i = c[0]; i < c[0] + 2; ++i)
            for (int j = c[1]; j < c[1] + 2; ++j)
                if (0 < i && i < m && 0 < j && j < n) {
                    res[cnt[i * 100000L + j]]--;
                    cnt[i * 100000L + j]++;
                    res[cnt[i * 100000L + j]]++;
                }
    return res;
}

bool checkArray(vector<int>& A, int k) {
    int cur = 0, n = A.size();
    for (int i = 0; i < n; ++i) {
        if (cur > A[i])
            return false;
        A[i] -= cur;
        cur += A[i];
        if (i >= k - 1)
            cur -= A[i - k + 1];
    }
    return cur == 0;
}

int sumOfSquares(vector<int>& nums) {
    int ans = 0;
    int n = nums.size();
    for(int i = 0; i < n; i++)

    return ans;
}
