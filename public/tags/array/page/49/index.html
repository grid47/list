Leetcode 1268: Search Suggestions System

You are given a perfect binary tree with n nodes, where each node has a cost associated with it. The tree is numbered from 1 to n, with node 1 as the root. For each node i, its left child is 2*i and its right child is 2*i + 1. You are allowed to increment the cost of any node by 1 any number of times. Your task is to return the minimum number of increments required to make the total cost of the path from the root to each leaf node equal.

int countSeniors(vector<string>& details) {
    int count = 0;
    for(auto i : details) {
        if(i[11] - '0' > 6) count++;
        else if (i[11] - '0' == 6 && i[12] - '0' > 0) count++;
    }
    return count;
}

int buyChoco(vector<int>& p, int m) {
    int l1 = 200, l2 = 200;
    for(int i = 0; i < p.size(); i++) {
        if(p[i] < l1) {
            l2 = l1;
            l1 = p[i];
        } else if(p[i] < l2) {
            l2 = p[i];
        }
    }
    
    if(l1 + l2 <= m) return m - l1 - l2;
    return m;
}

    /*
    
    0100111
    
    0000000 3 + 2 + 1        
    
    1011111 4 + 2 + 1   if odd len choose middle

    1011111 3 + 3 + 2 + 1         
    0100000 4 + 3 + 2 + 1 if odd len choose middle

    011011

            1/ 3 + 2 + 1
    
    010011
    
    010000  2 + 2 + 1
    
    101111  3
    011111  3 + 2
    111111  3 + 2 + 1 = 6
    
    010000  3 = 3
    100000  3 + 2 = 5
    000000  3 + 2 + 1 = 6
    
    */
    
    int n = s.size();
    
    if(n == 1) return 0;
    long long res = 0;

    char k = s[n / 2];
    
    int i, j;
    
    if(n % 2)
        i = n/2 + 1, j = n/2 - 1;
    else i = n/2, j = n / 2 - 1;
        bool ltog = true, rtog = true;
        while(i < n && j >= 0) {
            if(s[i] == k && rtog || s[i] != k && !rtog) {
            } else {
                rtog = !rtog;
                res += n - i;
            }
            i++;
            if((s[j] == k && ltog) || (s[j] != k && !ltog)) {
            } else {
                ltog = !ltog;
                res += j + 1;
            }
            j--;                
        }
        return res;
}

Observations:

int longestSemiRepetitiveSubstring(string s) {
    
    vector<int> sem(10, 0);
    int cnt = 0;
    
    int j = 0, len = 1;        
    for(int i = 1; i < s.size(); i++) {
        if(s[i] == s[i - 1]) {
            cnt++;
            sem[s[i] - '0']++;
        }
        while(cnt > 1 && j < i) {
            if(s[j] == s[j + 1]) {
                sem[s[j] - '0']--;
                cnt--;
            }
            j++;
        }
        len = max(len, i - j + 1);
    }
    return len;
}

int sumDistance(vector<int>& nums, string s, int d) {
    for(int i = 0; i < nums.size(); i++) {
        if(s[i] == 'L') nums[i] -= d;
        else nums[i] += d;
    }
    sort(nums.begin(), nums.end());
    int n = nums.size();
    long long ans = 0, dist = 0, mod = (int) 1e9 + 7;
    for(int i = 1; i < n; i++) {
        dist += ((long)nums[i] - nums[i - 1]) * (i);
        ans = (ans + dist) % mod;
    }
    return ans;
}
