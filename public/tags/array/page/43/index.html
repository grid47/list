Leetcode 1054: Distant Barcodes

You are given a list of distinct positive integers, nums, and a list of operations. Each operation consists of two elements: operations[i][0] and operations[i][1]. In each operation, you need to replace the element operations[i][0] in nums with the new value operations[i][1]. The number operations[i][1] is guaranteed to not already be in nums. After performing all the operations, return the final array.
Problem
Approach
Steps
Complexity

• Step 2: For each operation, find the index of the element to replace and update it with the new value.
• Step 3: After all operations are performed, return the final array.
Goal: The length of the array `nums` is at most 10^5, and each operation is performed at most 10^5 times.
Steps:
Assumptions:

Approach: We will approach this problem by maintaining a map that tracks the index of each number in `nums` to efficiently replace values during the operations.
Observations:
• The replacement operation involves direct lookup and update, so using a map can provide O(1) access time for each replacement.
• The algorithm will perform each operation in constant time, making the overall complexity O(m), where m is the number of operations.
Steps:
• Step 1: Create a map to store the indices of the elements in `nums`.
• Step 2: For each operation, find the index of the element to replace using the map, then replace the element in `nums` with the new value.
• Step 3: After processing all operations, return the updated array.
Empty Inputs:
• The input will never be empty as the number of elements in `nums` and the number of operations are both guaranteed to be at least 1.
Large Inputs:

vector<int> arrayChange(vector<int>& nums, vector<vector<int>>& ops) {
    
    map<int, int> mp;
    
    int n = nums.size();
    for(int i = 0; i < n; i++)
        mp[nums[i]] = i;
    
    for(auto it: ops) {
        int idx = mp[it[0]];
        mp.erase(it[0]);
        nums[idx] = it[1];
        mp[it[1]] = idx;
    }
    return nums;
}

    int mx = 0, sum = 0;
    for(int& a: A) {
        mx = max(a, mx);
        sum += a;
    }

int arithmeticTriplets(vector<int>& nums, int diff) {
int cnt[201] = {}, res = 0;
for (auto n : nums) {
    if (n >= 2 * diff)
        res += cnt[n - diff] && cnt[n - 2 * diff];

class Solution {
int ans;
public:
void solve(vector<vector<int>>& gph, vector<bool> &vis, int i) {
    vis[i] = true;
    for(int n: gph[i]) {
        if(!vis[n]) {
            ans++;
            vis[n] = true;
            solve(gph, vis, n);
        }
    }
}

int reachableNodes(int n, vector<vector<int>>& edges, vector<int>& restricted) {
    ans = 1;
    vector<vector<int>> gph(n);
    for(auto e: edges) {
        int u = e[0], v = e[1];
        gph[u].push_back(v);
        gph[v].push_back(u);
    }

    vector<bool> vis(n, false);
    for(auto i: restricted) vis[i] = true;

    solve(gph, vis, 0);
    return ans;
}

int longestIdealString(string s, int k) {
    int dp[150] = {}, res = 0;
    for(auto &i: s) {
        for(int j = i - k; j <= i + k; j++)
            dp[i] = max(dp[i], dp[j]);
        res = max(res, ++dp[i]);
    }
    return res;
}
