bool possibleBipartition(int n, vector<vector<int>>& dlk) {
    UF* uf = new UF(n + 1);
    vector<vector<int>> gph(n + 1);
    for(int i = 0; i < dlk.size(); i++) {
        gph[dlk[i][0]].push_back(dlk[i][1]);
        gph[dlk[i][1]].push_back(dlk[i][0]);            
    }

    for(int i = 1; i <= n; i++) {
        for(int x: gph[i]) {
            if(uf->find(x) == uf->find(i)) return false;
            uf->uni(gph[i][0], x);
        }
    }
    return true;
}

bool canReorderDoubled(vector<int>& arr) {
    
    unordered_map<int, int> c;
    for(int i : arr)
        c[i]++;
    
    vector<int> keys;
    for(auto it: c)
        keys.push_back(it.first);
    
    sort(keys.begin(), keys.end(), [&](int a, int b){
        return abs(a) < abs(b);
    });
    
    for(int x: keys) {
        if(c[x] > c[2 * x])
            return false;
        c[2 * x] -= c[x];
    }
    
    return true;
}

    if(!mp.count(key)) return "";

    int l = 0, r = mp[key].size() - 1;
    while(l <= r) {

    }
