int fib(int n) {
    if(n == 0) return 0;
    vector<int> dp(n + 1, 0);
    dp[0] = 0;
    dp[1] = 1;
    
    for(int i = 2; i <= n; i++)
    dp[i] = dp[i - 1] + dp[i - 2];
        
    return dp[n];
}

string findLongestWord(string s, vector<string>& d) {
    
    string ans;
    for(int i = 0; i < d.size(); i++) {
        
        int pi = 0, pj = 0;
        
        for(; pi < s.size() && pj < d[i].size(); pi++) {
            pj += s[pi] == d[i][pj];
        }
        
        if(pj == d[i].size() && (ans.size() < d[i].size() || (ans.size() == d[i].size() && ans > d[i])))
            ans = d[i];
        
    }
    return ans;
}

• This problem can be solved using a technique based on prefix sums or difference arrays.

Steps:
• 1. Initialize a score variable to 0 and a hash map to store the first occurrence of each score.
• 2. Traverse through the array. For each element, update the score by adding 1 for a 1 or subtracting 1 for a 0.
• 3. If the score is 0, update the maximum subarray length as the current index + 1.
• 4. If the score is already in the hash map, calculate the length of the subarray from the first occurrence of the score to the current index and update the maximum length.

int findMaxLength(vector<int>& nums) {

    int n = nums.size();
    int score = 0, res = 0;
    map<int, int> mp;

    for(int i = 0; i < n; i++) {
        score += nums[i]? 1: -1;

        if(score == 0) res = i + 1;
        else {

            if(mp.count(score))
            res = max(res, i - mp[score]);

            if(!mp.count(score)) mp[score] = i;
        }
    }
    return res;
}

• Compare the squares of the elements at the two pointers and place the larger square at the correct position in the result array (starting from the last position).

• Return the result array once the two pointers have processed all elements.

vector<int> sortedSquares(vector<int>& nums) {
    vector<int> ans(nums.size());
    int l = 0, r = nums.size() - 1;
    int ll, rr;
    for(int i = nums.size() - 1; i >= 0; i--) {
        ll = nums[l] * nums[l];
        rr = nums[r] * nums[r];
        if(ll < rr) ans[i] = rr, r--;
        else ans[i] = ll, l++;
    }
    return ans;
}
