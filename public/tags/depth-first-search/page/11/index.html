Leetcode 756: Pyramid Transition Matrix

Your task is to determine the path of each ball dropped from the top of the box. The ball can either fall out of the bottom, or get stuck if it hits a ‘V’ shaped pattern between two boards or is redirected into a wall.

vector<int> findBall(vector<vector<int>>& grid) {
    vector<int> res;
    int m = grid.size(), n = grid[0].size();
    for(int i = 0; i < n; i++) {
        int i1 = i, i2;
        for(int j = 0; j < m; j++) {
            i2 = i1 + grid[j][i1];
            if(i2 <0 || i2 >= n || grid[j][i1] != grid[j][i2]) {
                i1 = -1;
                break;
            }
            i1 = i2;
        }
        res.push_back(i1);
    }
    return res;
}

string longestNiceSubstring(string s, int start = 0, int end = -1) {
    if (end == -1)
        end = s.size();
    int cnt[26][2] = {}, j = start - 1;
    for (auto i = start; i < end; ++i)
        cnt[tolower(s[i]) - 'a'][(bool)islower(s[i])] = 1;
    string res;
    for (auto i = start; i <= end; ++i) {
        int ch = i == end ? -1 : tolower(s[i]) - 'a';
        if (ch == -1 || cnt[ch][0] + cnt[ch][1] == 1) {
            if (j == -1 && ch == -1)
                return s;
            auto res1 = longestNiceSubstring(s.substr(j + 1, i - j - 1));
            if (res1.size() > res.size())
                res = res1;
            j = i;
        }
    }
    return res;
}

Leetcode 787: Cheapest Flights Within K Stops

The goal is to maximize the height values of land cells while ensuring the height of water cells is 0. Return the grid with assigned heights.

Approach
Steps
Complexity

Example: isWater = [[0,0,1],[1,0,0],[0,0,0]]

• 1 <= m, n <= 1000
• isWater[i][j] is 0 or 1

vector<vector<int>> highestPeak(vector<vector<int>>& isWater) {
    int m = isWater.size(), n = isWater[0].size();
    
    vector<vector<int>> ans(m, vector<int>(n, 0)), vis(m, vector<int>(n, 0));
    
    queue<vector<int>> q;
    for(int i = 0; i < m; i++)
    for(int j = 0; j < n; j++)
        if(isWater[i][j] == 1)
            q.push({i, j});
    
    int rot[] = {0, 1, 0, -1, 0};
    
    int cur = 0;
    while(!q.empty()) {
        int sz = q.size();
        while(sz--) {
            auto it = q.front();
            q.pop();
            if(vis[it[0]][it[1]]) continue;
            vis[it[0]][it[1]] = 1;
            
            ans[it[0]][it[1]] = cur;
            
            for(int i = 0; i < 4; i++) {
                int x = it[0] + rot[i], y = it[1] + rot[i + 1];
                if(x < 0 || y < 0 || x == m || y == n || vis[x][y]) continue;
                q.push({x, y});
            }
        }
        cur++;
    }
    return ans;
}
