int numIslands(vector<vector<char>>& grid) {
    int cnt = 0;
    int m = grid.size(), n = grid[0].size();
    for(int i = 0; i < m; i++)
    for(int j = 0; j < n; j++)
        if(grid[i][j] == '1') {
            cnt++;
            dfs(grid, i, j);
        }
    return cnt;
}
void dfs(vector<vector<char>> &grid, int i, int j) {
    if(i < 0 || j < 0 || i == grid.size() || j == grid[0].size() || grid[i][j] == '0')
        return ;
    grid[i][j] = '0';
    dfs(grid, i + 1, j);
    dfs(grid, i, j + 1);
    dfs(grid, i - 1, j);
    dfs(grid, i, j - 1);
}

Leetcode 113: Path Sum II

int sqr(int n) {
    int res = 0;
    while(n) {
        int tmp = n % 10;
        res += tmp * tmp;
        n = n / 10;
    }
    return res;
}
bool isHappy(int n) {
    int slow = n;
    int fast = n;
    do {
        slow = sqr(slow);
        fast = sqr(sqr(fast));
    } while(slow != fast);
    return slow == 1;
}

bool isIsomorphic(string s, string t) {
   
     
    map<char, char> fwd, rwd;
    int n = s.size();
    for(int i = 0; i < n; i++) {
        if (fwd.count(s[i])){
            if(fwd[s[i]] != t[i])
                return false;
        }
        if(rwd.count(t[i])){
            if(rwd[t[i]] != s[i])
                return false;
        }
        fwd[s[i]] = t[i];
        rwd[t[i]] = s[i];
    }
    return true;
}
