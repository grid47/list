Leetcode 385: Mini Parser

Your task is to determine if you can provide the correct change to every customer as they arrive in line. bool lemonadeChange(vector<int>& bills) {
    int five = 0, ten = 0;
    for (int x : bills) {
             if (x == 5) five++;
        else if (x == 10) {
                 if (five > 0) five--, ten++;
            else return false;
        } else {
                 if (five > 0 && ten > 0) five--, ten--;
            else if (five > 2) five -= 3;
            else return false;
        }
    }
    return true;
}

Leetcode 386: Lexicographical Numbers

The matrix’s score is calculated by interpreting each row as a binary number and summing up these values. Your task is to determine the maximum score possible after performing any number of toggle operations.

int matrixScore(vector<vector<int>>& grid) {
    int m = grid.size(), n = grid[0].size();
    int res = (1 << n - 1) * m;

    for(int j = 1; j < n; j++) {
        int cur = 0;
        for(int i = 0; i < m; i++)
            cur += grid[i][0] == grid[i][j];
        res += max(cur, m - cur) * (1<<n-j-1);
    }

    return res;
}

pair<TreeNode*, int> lcadn(TreeNode* root) {
    if(root == NULL) return {NULL, 0};
    auto lft = lcadn(root->left);
    auto rht = lcadn(root->right);
    if(lft.second < rht.second) {
        return {rht.first, rht.second + 1};
    } else if (lft.second > rht.second) {
        return {lft.first, lft.second + 1};            
    }
    return {root, lft.second + 1};
}

int maxSubarraySumCircular(vector<int>& nums) {
    int lmax = nums[0], lmin = nums[0];
    int gmax = nums[0], gmin = nums[0];
    int sum = nums[0];
    for(int i = 1; i < nums.size(); i++) {
        sum += nums[i];
        lmax = max(nums[i], lmax + nums[i]);
        gmax = max(lmax, gmax);
        lmin = min(nums[i], lmin + nums[i]);
        gmin = min(lmin, gmin);
    }
    return gmax > 0? max(gmax, sum - gmin): gmax;
}
