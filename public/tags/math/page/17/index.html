Leetcode 1232: Check If It Is a Straight Line

long long findScore(vector<int>& nums) {
    long long score = 0;
    priority_queue<vector<int>, vector<vector<int>>, greater<vector<int>>> pq;
    int n = nums.size();
    unordered_map<int, int> mp;
    for(int i = 0; i < n; i++)
        pq.push({nums[i], i});
    
    while(!pq.empty()) {
        while(!pq.empty() && mp.count(pq.top()[1])) pq.pop();
        if(!pq.empty()) {
            score += pq.top()[0];
            mp[pq.top()[1]] = true;
            mp[pq.top()[1] + 1] = true;
            mp[pq.top()[1] - 1] = true;                
            pq.pop();
        }
    }
    return score;
}

int findSmallestInteger(vector<int>& nums, int value) {
    map<int, int> mp;
    int n = nums.size();
    for(int i = 0; i < n; i++) {
        mp[(nums[i] < 0? (nums[i] % value + value)%value: nums[i]%value)]++;
    }
    int idx = 0, mn = INT_MAX;
    for(int i = 0; i < value; i++) {
        if(mp[i] < mn) {
            idx = i;
            mn = mp[i];
        }
    }
    return mn * value + idx;
}

int kItemsWithMaximumSum(int one, int zero, int neg, int k) {
    int sum = 0;
    sum += min(one, k);
    k -= min(one, k);
    if(k > 0) {
        k -= min(zero, k);
    }
    if(k > 0) {
        sum -= min(neg, k);
        k -= min(neg, k);
    }
    return sum;
}

    unordered_map<int, double> mp, cnt;
    int n = nums.size();
    vector<long long> arr(n, 0);
    for(int i = 0; i < n; i++) {
        arr[i] = cnt[nums[i]] * i - mp[nums[i]];
        mp[nums[i]] += i;
        cnt[nums[i]] += 1;
    }
    mp.clear(), cnt.clear();        
    for(int i = n - 1; i >= 0; i--) {
        arr[i] += mp[nums[i]] - cnt[nums[i]] * i;
        mp[nums[i]] += i;
        cnt[nums[i]] += 1;
    }
    return arr;
}

vector<int> colorTheArray(int n, vector<vector<int>>& queries) {
int c[100003] = {}, adj = 0;
vector<int> res;
for (const auto &q : queries) {
    int i = q[0] + 1, color_i = q[1];
    if (c[i])
        adj -= (c[i - 1] == c[i]) + (c[i + 1] == c[i]);
    c[i] = color_i;
    adj += (c[i - 1] == c[i]) + (c[i + 1] == c[i]);
    res.push_back(adj);
}
return res;
}
