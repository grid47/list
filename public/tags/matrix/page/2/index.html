    this->preorder = preorder;
    this->inorder = inorder;

    for(int i = 0; i < inorder.size(); i++) mp[inorder[i]] = i;

    return helper(0, 0, inorder.size() - 1);

TreeNode* helper(int ps, int is, int ie) {
    if(ps >= preorder.size() || is > ie) return NULL;

    TreeNode* root = new TreeNode(preorder[ps]);
    root->left  = helper(ps + 1, is, mp[root->val] - 1);
    root->right = helper(ps + mp[root->val] - is + 1, mp[root->val] + 1, ie);

    return root;
}

vector<vector<int>> levelOrderBottom(TreeNode* root) {

    vector<vector<int>> ans;        
    queue<TreeNode*> q;
    if(root == NULL) return ans;
    
    q.push(root);        
    stack<vector<int>> stk;
            
    while(!q.empty())
    {
        vector<int> ans;
        int sz = q.size();

        while(sz--) {
            TreeNode* tmp = q.front();
            q.pop();
            ans.push_back(tmp->val);

            if(tmp->left)   q.push(tmp->left);
            if(tmp->right)  q.push(tmp->right);                
        }
        stk.push(ans);
    }

    while(!stk.empty())
    {
        ans.push_back(stk.top());
        stk.pop();
    }

    return ans;      
}

void flatten(TreeNode* root) {

    if( root == NULL) return;

    TreeNode* tmp = root->right;
    root->right = root->left;
    root->left = nullptr;

    TreeNode* node = root;
    while(node->right) node = node->right;

    node->right = tmp;
    flatten(root->right); 

}

int dfs(int i, vector<int> &quiet) {
    if(res[i] >= 0) return res[i];
    res[i] = i;
    for(int j : richer2[i])
        if(quiet[res[i]] > quiet[dfs(j, quiet)]) res[i] = res[j];
    return res[i];
}
