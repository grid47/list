class Solution { public: vector> grid, memo; int m, n; bool isPossibleToCutPath(vector>& grid) { m = grid.size(); n = grid[0].size(); this->grid= grid; if(isconnected(0, 0) == false) return true; this->grid[0][0] = 1; return !isconnected(0, 0); } bool isconnected(int i, int j) { if(i == m - 1 && j == n - 1) return true; if(i >= m || j >= n || grid[i][j] == 0) return false; grid[i][j] = 0; return isconnected(i + 1, j) || isconnected(i, j + 1); } };
