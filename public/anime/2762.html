class Solution { public: long long continuousSubarrays(vector& nums) { long long res = 0; int i = 0, j = 0,n=nums.size(); map mp; while(j3 || abs(mp.begin()->first - mp.rbegin()->first) > 2) { mp[nums[i]]--; if(mp[nums[i]] == 0) mp.erase(nums[i]); i++; } long long temp = j-i+1; res = res + temp; j++; } return res; } };
