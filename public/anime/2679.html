class Solution { public: int matrixSum(vector>& nums) { int res = 0; int m = nums.size(), n = nums[0].size(); for(int i = 0; i < m; i++) sort(nums[i].rbegin(), nums[i].rend()); for(int i = 0; i < n; i++) { int mx = nums[0][i]; for(int j = 0; j < m; j++) { mx = max(mx, nums[j][i]); } res += mx; } return res; } };
