class Solution { public: int minSubarray(vector& nums, int p) { map mp; mp[0] = -1; int n = nums.size(), res = n, cur = 0, need = 0, want = 0; for(int a : nums) need = (need + a) % p; for(int i = 0; i < n; i++) { cur = (cur + nums[i]) % p; mp[cur] = i; want = (cur - need + p) % p; if (mp.count(want)) res = min(res, i - mp[want]); } return res < n? res : -1; } };
