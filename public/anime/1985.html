class cmp { public: bool operator () (string &b, string &a) { if (a.size() != b.size()) return a.length() < b.length(); return a < b; } }; class Solution { public: string kthLargestNumber(vector& nums, int k) { priority_queue, cmp> pq; for(int i = 0; i < nums.size(); i++) { pq.push(nums[i]); while(pq.size() > k) { pq.pop(); } } return pq.top(); } };
