class Solution { public: vector> displayTable(vector>& orders) { vector> table(501); set s; for(vector &v : orders) { s.insert(v[2]); ++table[stoi(v[1])][v[2]]; } vector> res; for(int t = 0; t < 501; t++) { if (t > 0 && table[t].empty()) { continue; } res.push_back(vector()); res.back().push_back(t == 0 ? "Table" : to_string(t)); for(auto it = begin(s); it != end(s); it++) { res.back().push_back(t == 0? *it : to_string(table[t][*it])); } } return res; } };
