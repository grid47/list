class Solution { public: int evalRPN(vector& tokens) { stack stk; int n = tokens.size(); for(int i = 0; i < n; i++) { if(tokens[i] != "+" && tokens[i] != "-" && tokens[i] != "*" && tokens[i] != "/") { stk.push(stoi(tokens[i])); } else { int x = stk.top(); stk.pop(); int y = stk.top(); stk.pop(); if(tokens[i] == "*"){ stk.push(y * x); } if(tokens[i] == "/"){ stk.push(y / x); } if(tokens[i] == "+"){ stk.push(y + x); } if(tokens[i] == "-"){ stk.push(y - x); } } } return stk.top(); } };
