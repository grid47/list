class Solution { public: vector findIndices(vector& nums, int idif, int vdif) { int mn = 0, mx = 0; for (int i = idif, j = 0; i < nums.size(); ++i, ++j) { mn = nums[mn] < nums[j] ? mn : j; mx = nums[mx] > nums[j] ? mx : j; if (abs(nums[i] - nums[mn]) >= vdif) return {mn, i}; if (abs(nums[i] - nums[mx]) >= vdif) return {mx, i}; } return {-1, -1}; } };
