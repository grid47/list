class Solution { public: int maxAreaOfIsland(vector>& grid) { int mx = 0; int m = grid.size(), n = grid[0].size(); for(int i = 0; i < m; i++) for(int j = 0; j < n; j++) if(grid[i][j] == 1) mx = max(mx, dfs(grid, i, j)); return mx; } int dfs(vector>& grid, int i, int j) { int m = grid.size(), n = grid[0].size(); if(i < 0 || j < 0 || i == m || j == n || grid[i][j] != 1) return 0; int ans = 1; grid[i][j] = 2; ans += dfs(grid, i + 1, j); ans += dfs(grid, i, j + 1); ans += dfs(grid, i - 1, j); ans += dfs(grid, i, j - 1); return ans; } };
