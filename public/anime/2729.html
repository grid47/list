class Solution { public: bool isFascinating(int n) { int n1 = 2 * n; int n2 = 3 * n; vector yes(10, false); string s = to_string(n) + to_string(n1) + to_string(n2); for(int i = 0; i < s.size(); i++) { if(s[i] == '0') return false; if(yes[s[i] - '0']) return false; yes[s[i] - '0'] = true; } return true; } };
