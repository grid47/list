class Solution { public: vector> floodFill(vector>& image, int sr, int sc, int color) { if(color == image[sr][sc]) return image; dfs(image, sr, sc, color, image[sr][sc]); return image; } void dfs(vector> & img, int r, int c, int color, int node) { int m = img.size(), n = img[0].size(); if (r < 0 || c < 0 || r == m || c == n || img[r][c] != node) return; img[r][c] = color; dfs(img, r + 1, c, color, node); dfs(img, r, c + 1, color, node); dfs(img, r - 1, c, color, node); dfs(img, r, c - 1, color, node); } };
