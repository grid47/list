class Solution { public: vector cnt; int no(int x) { int sum = 0; for(int i = 0; i <= 50; i++) { sum += cnt[i]; if(sum >= x) return i - 50; } return 0; } vector getSubarrayBeauty(vector& nums, int k, int x) { cnt.resize(101, 0); vector ans; int n = nums.size(); for(int i = 0; i < k; i++) { cnt[nums[i] + 50]++; } ans.push_back(no(x)); for(int i = k; i < n; i++) { cnt[nums[i] + 50]++; cnt[nums[i - k] + 50]--; ans.push_back(no(x)); } return ans; } };
