class Solution { public: int minInsertions(string s) { int res = 0; int n = s.size(); stack stk; for(int i = 0; i < n; i++) { if(s[i] == '(') { stk.push('('); continue; } if(i + 1 < n && s[i + 1] == ')') { if(!stk.empty()) stk.pop(); else res++; i++; } else { if(!stk.empty()) {stk.pop(); res++;} else res += 2; } } return res + stk.size() * 2; } };
