class Solution { public: vector> queensAttacktheKing(vector>& queens, vector& k) { vector> b(8, vector(8, false)); for(auto q: queens) b[q[0]][q[1]] = true; vector> res; for(int i = -1; i <= 1; i++) for(int j = -1; j <= 1; j++) { if ( i == 0 && j == 0) continue; int x = k[0] + i, y = k[1] + j; while(min(x, y) >= 0 && max(x, y) < 8) { if(b[x][y]) { res.push_back({x, y}); break; } x += i; y += j; } } return res; } };
