class Solution { public: int twoCitySchedCost(vector>& costs) { /* 2n people - interview */ int mc = 0; vector refund; for(auto c : costs) { mc += c[0]; refund.push_back(c[1] - c[0]); } sort(refund.begin(), refund.end()); int n = costs.size()/ 2; for(int i = 0; i < n; i++) { mc += refund[i]; } return mc; } };
