class Solution { public: int minImpossibleOR(vector& nums) { set cnt; int n = nums.size(); for(int i = 0; i < n; i++) { cnt.insert(nums[i]); } int ret = 1; while(cnt.count(ret)) { ret *= 2; } // to create all the numbers till a positive integer // presents of all the powers of 2s below it are mandatory. return ret; } };
