class Solution { public: int findMaxLength(vector& nums) { int n = nums.size(); int score = 0, res = 0; map mp; for(int i = 0; i < n; i++) { score += nums[i]? 1: -1; if(score == 0) res = i + 1; else { if(mp.count(score)) res = max(res, i - mp[score]); if(!mp.count(score)) mp[score] = i; } } return res; } };
