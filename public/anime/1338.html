class Solution { public: int minSetSize(vector& arr) { unordered_map mp; for(int c: arr) ++mp[c]; vector frq; for(auto [_, fq] : mp) frq.push_back(fq); sort(frq.begin(), frq.end()); int ans = 0, i = frq.size() - 1, half = arr.size()/2, rm = 0; while(rm < half) { rm += frq[i--]; ans++; } return ans; } };
