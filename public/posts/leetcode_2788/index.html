Leetcode 2788: Split Strings by Separator

Easy
Yasir

Code:

class Solution {
public:
    vector<string> splitWordsBySeparator(vector<string>& words, char sep) {
        
        vector<string> ans;
        
        for(int i = 0; i < words.size(); i++) {
            string cur = "";
            
            for(int j = 0; j < words[i].size(); j++) {
                if(words[i][j] == sep) {
                    if(cur != "")
                    ans.push_back(cur);
                    cur = "";
                } else if(j == words[i].size() - 1) {
                    cur += words[i][j];                    
                    ans.push_back(cur);
                    cur = "";
                } else 
                cur += words[i][j];
            }
            
        }
        return ans;
    }
};
