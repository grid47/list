{{< highlight cpp >}}
class Solution {
public:
    int rev(int num) {
        int res = 0;
        while(num > 0) {
            res = res * 10 + num % 10;
            num /= 10;
        }
        return res;
    }
    
    int countDistinctIntegers(vector<int>& nums) {
        set<int> cnt;
        int n = nums.size();
        for(int i = 0; i < n; i++) {
            cnt.insert(nums[i]);
            cnt.insert(rev(nums[i]));
        }
        
        return cnt.size();        
    }
};
