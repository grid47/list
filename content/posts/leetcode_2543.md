class Solution {
public:
    bool isReachable(int x, int y) {
        if(x == 1 && y == 1) return true;

        bool ans = false;
        if(y%2 == 0) return isReachable(x, y/2);        
        if(x%2 == 0) return isReachable(x/2, y);        
        if(x > y)    return isReachable(x - y, y);
        if(x < y)    return isReachable(x, y - x);        
        return ans;
    }
};
