**Code:**

{{< highlight cpp >}}
SELECT v.customer_id, COUNT(v.visit_id) AS count_no_trans 
from Visits v 
LEFT JOIN Transactions t 
ON v.visit_id = t.visit_id  
WHERE t.amount IS NULL  and t.transaction_id IS NULL
GROUP BY v.customer_id; 
{{< /highlight >}}
