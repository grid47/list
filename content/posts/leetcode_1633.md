**Code:**

{{< highlight cpp >}}
# Write your MySQL query statement below
SELECT contest_id, IFNULL(ROUND(COUNT(DISTINCT user_id) * 100/(SELECT COUNT(user_id) FROM Users), 2), 0) percentage
FROM Register
GROUP BY contest_id
ORDER BY percentage DESC, contest_id;
