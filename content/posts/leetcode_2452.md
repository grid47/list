class Solution {
public:
    vector<string> twoEditWords(vector<string>& queries, vector<string>& dictionary) {
        vector<string> res;
        for(auto &q: queries)
        for(auto &d: dictionary)
        if(inner_product(begin(q), end(q), begin(d), 0, plus<int>(), not_equal_to<char>()) < 3) {
            res.push_back(q);
            break;
        }
        return res;
    }
};
