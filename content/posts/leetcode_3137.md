class Solution {
public:
    int minimumOperationsToMakeKPeriodic(string s, int k) {
        
        int n = s.size();
        map<string, int> mp;
        
        for(int i = 0; i < n / k; i++) {
            mp[s.substr(i * k, k)]++;
        }
        
        int lg = 0, net = 0;
        for(auto it: mp) {
            lg = max(lg, it.second);
            net += it.second;
        }
        
        return net - lg;
        
    }
};
