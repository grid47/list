class Solution {
public:
    vector<int> findThePrefixCommonArray(vector<int>& A, vector<int>& B) {
        set<int> a, b, c;
        int n = A.size();
        vector<int> ans(n, 0);
        for(int i = 0; i < n; i++) {
            if(a.count(B[i])) {
                c.insert(B[i]);
            } 
            if(b.count(A[i])) {
                c.insert(A[i]);                
            } 
            if(A[i] == B[i]) {
                c.insert(A[i]);                
            }
            a.insert(A[i]);
            b.insert(B[i]);
            ans[i] = c.size();
        }
        return ans;
    }
};
