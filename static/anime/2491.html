class Solution { public: long long dividePlayers(vector& skill) { sort(skill.begin(), skill.end()); int n = skill.size(); int l = 0, r = n - 1; int sum = skill[l] + skill[r]; long long prod = skill[l] * skill[r]; l++, r--; while(l < r) { if(sum != (skill[l] + skill[r])) return -1; prod += skill[l] * skill[r]; l++; r--; } return prod; } };
