class Solution { public: bool isPossibleDivide(vector& nums, int k) { map cnt; int n = nums.size(); for(int num : nums) cnt[num]++; for(auto it : cnt) { int frq = it.second; if(frq > 0) for(int i = 0; i < k; i++) { if(cnt[it.first + i] < frq) return false; else cnt[it.first + i] -= frq; } } return true; } };
