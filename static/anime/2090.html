class Solution { public: vector getAverages(vector& nums, int k) { int n = nums.size(); vector psum(n, 0); vector avg(n, 0); psum[0] = nums[0]; for(int i = 1; i < n; i++) psum[i] = psum[i - 1] + nums[i]; for(int i = 0; i < n; i++) { if(i - k < 0 || i + k >= n) { avg[i] = -1; } else { avg[i] = (psum[i + k] - ((i - k) == 0? 0: psum[i - k - 1])) / (2 * k + 1); } } return avg; } };
