class Solution { vector mem; public: int combinationSum4(vector& nums, int target) { mem.resize(target + 1, -1); return dp(nums, target); } int dp(vector& nums, int left) { if(left == 0) return 1; if(left < 0) return 0; if(mem[left] != -1) return mem[left]; int res = 0; for(int i = 0; i < nums.size(); i++) res += dp(nums, left - nums[i]); return mem[left] = res; } };
