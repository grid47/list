class Solution { public: vector findIntersectionValues(vector& nums1, vector& nums2) { set n1, n2; for(int x: nums1) n1.insert(x); for(int x: nums2) n2.insert(x); int cnt1 = 0; for(int x: nums1) if(n2.count(x)) cnt1++; int cnt2 = 0; for(int x: nums2) if(n1.count(x)) cnt2++; return vector{cnt1, cnt2}; } };
