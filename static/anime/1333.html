class Solution { public: static bool cmp(vector &a, vector &b) { if(a[1] == b[1]) return a[0] > b[0]; else return a[1] > b[1]; } vector filterRestaurants(vector>& rest, int veg, int mxp, int mxd) { vector> ans; for(auto r: rest) { if((!veg || (veg == r[2])) && r[3] <= mxp && r[4] <= mxd) { ans.push_back({r[0], r[1]}); } } sort(ans.begin(), ans.end(), cmp); vector res; for(auto x: ans) { res.push_back(x[0]); } return res; } };
