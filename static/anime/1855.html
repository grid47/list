class Solution { public: int maxDistance(vector& nums1, vector& nums2) { int ans = 0; for(int i = 0; i < nums1.size(); i++) { int l = i, r = nums2.size() - 1, j = i; int target = nums1[i]; // find larget index which satisfies target <= nums[i] while(l <= r) { int mid = l + (r - l + 1) / 2; if(target < nums2[mid]) { j = mid; l = mid + 1; } else if(target > nums2[mid]) { r = mid - 1; } else if(target == nums2[mid]) { j = mid; l = mid + 1; } } // cout << j - i; ans = max(ans, j - i); } return ans; } };
