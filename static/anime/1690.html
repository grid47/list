class Solution { public: int memo[1001][1001] = {}; int dp(vector& s, int i, int j, int sum) { if(i == j) { return 0; } return memo[i][j] ? memo[i][j] : memo[i][j] = max(sum - s[i] - dp(s, i + 1, j, sum - s[i]), sum - s[j] - dp(s, i, j - 1, sum - s[j])); } int stoneGameVII(vector& s) { int n = s.size(); return dp(s, 0,n-1, accumulate(begin(s), end(s), 0)); } };
