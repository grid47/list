class Solution { public: int longestWPI(vector& hours) { int n = hours.size(), res = 0, score = 0; map mp; for(int i = 0; i < n; i++) { score += (hours[i] > 8) ? 1 : -1; if(score > 0)res = i + 1; else { if (!mp.count(score)) mp[score] = i; if (mp.count(score - 1)) res = max(res, i - mp[score -1]); } } return res; } };
