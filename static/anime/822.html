class Solution { public: int flipgame(vector& fronts, vector& backs) { unordered_set same; for(int i = 0; i < fronts.size(); i++) if(fronts[i] == backs[i]) same.insert(fronts[i]); int res = 3000; for(auto &num: fronts) if(!same.count(num)) res = min(res, num); for(auto &num: backs) if(!same.count(num)) res = min(res, num); return res % 3000; } };
