class Solution { public: int matchPlayersAndTrainers(vector& man, vector& master) { priority_queue, greater> pq, qq; int m = man.size(), n = master.size(); for(int i = 0; i < m; i++) { pq.push(man[i]); } for(int i = 0; i < n; i++) { qq.push(master[i]); } int cnt = 0; while(!pq.empty() && !qq.empty()) { while(!qq.empty() && pq.top() > qq.top()) qq.pop(); if(!qq.empty()) { cnt++; qq.pop(); pq.pop(); } } return cnt; } };
