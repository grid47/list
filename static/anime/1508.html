class Solution { public: int mod = (int) 1e9 + 7; int rangeSum(vector& nums, int n, int left, int right) { // binary search vector ans; for(int i = 0; i < n; i++) { int sum = 0; for(int j = i; j < n; j++) { sum = (sum + nums[j]) % mod; ans.push_back(sum); } } sort(ans.begin(), ans.end()); int res = 0; for(int i = left - 1; i < right; i++) res = (res + ans[i]) % mod; return res; } };
