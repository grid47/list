class Solution { public: long long minCost(vector& b1, vector& b2) { map c1; for(int x: b1) c1[x]++; for(int x: b2) c1[x]--; long long res = 0; vector swaps; for(auto [k1, v1]: c1){ if(v1 % 2) return -1; for(int i = 0; i < abs(v1) / 2; i++) swaps.push_back(k1); } int mn = c1.begin()->first; for(int i = 0; i < swaps.size() / 2; i++) res += min(swaps[i], mn * 2); return res; } };
