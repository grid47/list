class Solution { public: int countNicePairs(vector& nums) { unordered_map mp; for(auto &n: nums) mp[n - rev(n)]++; int mod = 1000000007; long count = 0; for(auto &it: mp) count = (count + (it.second * (it.second - 1))/ 2) % mod; return count; } int rev(int x) { int revNum = 0; while(x) revNum = revNum * 10 + (x%10), x /= 10; return revNum; } };
