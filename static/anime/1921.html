class Solution { public: int eliminateMaximum(vector& dist, vector& speed) { int n = dist.size(); vector res(n, 0); for(int i = 0; i < n; i++) res[i] = (double(dist[i]) / speed[i]); sort(res.begin(), res.end()); long j = 0, ans = 0; for (int i = 0; i < n; i++) if(j < res[i]) { ans++; j++; } else break; return (int) ans; } };
