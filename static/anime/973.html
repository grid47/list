class cmp { public: bool operator()(vector &a, vector &b) { return a[0] * a[0] + a[1] * a[1] > b[0]* b[0] + b[1] * b[1]; } }; class Solution { public: vector> kClosest(vector>& pts, int k) { priority_queue, vector>, cmp> pq; for(auto it : pts) { pq.push(it); } vector> ans; while(k--) { ans.push_back(pq.top()); pq.pop(); } return ans; } };
