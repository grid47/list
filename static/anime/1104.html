class Solution { public: vector pathInZigZagTree(int label) { vector res; res.push_back(label); int level = log2(label); while (level != 0) { int left = pow(2 , (level - 1)); int right = pow(2 , level) - 1; label = left + ( right- label / 2); res.push_back(label); level -= 1; } int num = res.size(); reverse(res.begin(), res.end()); return res; } };
