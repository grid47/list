class Solution { public: vector prisonAfterNDays(vector& cells, int n) { set> seen; int i = 0, flag = 0; vector c; for(i = 0; i < n; i++) { vector v = getNext(cells); if(!seen.count(v)) { seen.insert(v); cells = v; } else { for(int j = 0; j < n % i; j++) { cells = getNext(cells); } return cells; } } return cells; } vector getNext(vector &cells) { vector n(cells.size(), 0); for(int i = 1; i < cells.size() -1; i++) n[i] = cells[i-1] == cells[i+1]?1:0; return n; } };
