class Solution { public: vector commonChars(vector& words) { vector frq(26, INT_MAX); for(string& s: words) { vector cnt(26, 0); for(char x: s) cnt[x - 'a']++; for(int i = 0; i < 26; i++) frq[i] = min(frq[i], cnt[i]); } vector res; for(int i = 0; i < 26; i++) for(int j = 0; j < frq[i]; j++) res.push_back(string(1, 'a' + i)); return res; } };
